82void mpi_server_dispatcher(struct st_th th)
83{
84 int ret;
85
86 // check params...
87 if (NULL == th.params)
88 {
89 printf("[WORKERS ID=%d] ERROR: NULL arguments", th.id);
90 return;
91 }
92
93 struct st_th th_arg;
94 int disconnect = 0;
95 while (!disconnect)
96 {
97 ret = mpi_server_comm_read_operation(th.params, (MPI_Comm)th.sd, (char *)&(th.type_op), 1, &(th.rank_client_id));
98 if (ret < 0)
99 {
100 debug_info("[MPI-SERVER] ERROR: mpi_server_comm_readdata fail\n");
101 return;
102 }
103
104 if (th.type_op == MPI_SERVER_DISCONNECT)
105 {
106 debug_info("[MPI-SERVER] INFO: DISCONNECT received\n");
107 disconnect = 1;
108 continue;
109 }
110
111 if (th.type_op == MPI_SERVER_FINALIZE)
112 {
113 debug_info("[MPI-SERVER] INFO: FINALIZE received\n");
114 disconnect = 1;
115 the_end = 1;
116 continue;
117 }
118
119 // Launch worker per operation
120 th_arg.params = &params;
121 th_arg.sd = (MPI_Comm)th.sd;
122 th_arg.function = mpi_server_run;
123 th_arg.type_op = th.type_op;
124 th_arg.rank_client_id = th.rank_client_id;
125 th_arg.wait4me = FALSE;
126
127 workers_launch(&worker, &th_arg, mpi_server_run);
128 }
129
130 debug_info("[MPI-SERVER] mpi_server_worker_run (ID=%d) close\n", th.rank_client_id);
131
132 mpiClient_comm_close((MPI_Comm)th.sd);
133}

239int mpi_server_down(int argc, char *argv[])
240{
241 int ret, buf;
242 char port_name[MPI_MAX_PORT_NAME];
243 char srv_name[1024];
244 char dns_name[2048];
245 MPI_Comm server;
246 FILE *file;
247
248 printf("\n");
249 printf(" ----------------\n");
250 printf(" Stopping servers (%s)\n", serv_name);
251 printf(" ----------------\n");
252 printf("\n");
253
254 MPI_Init(&argc, &argv);
255
256 // Open host file
257 file = fopen(params.shutdown_file, "r");
258 if (file == NULL)
259 {
260 printf("[MPI-SERVER] ERROR: invalid file %s\n", params.shutdown_file);
261 return -1;
262 }
263
264 while (fscanf(file, "%[^\n] ", srv_name) != EOF)
265 {
266 int version_len;
267 char version[MPI_MAX_LIBRARY_VERSION_STRING];
268 MPI_Get_library_version(version, &version_len);
269
270 if (strncasecmp(version, "Open MPI", strlen("Open MPI")) != 0)
271 {
272 // Lookup port name
273 char aux_srv_ip[1024];
274 ret = ns_lookup("mpi_server", srv_name, aux_srv_ip, port_name);
275 if (ret < 0)
276 {
277 printf("[MPI-SERVER] ERROR: server %s not found\n", dns_name);
278 continue;
279 }
280 }
281 else
282 {
283 // Lookup port name on nameserver
284 ret = MPI_Lookup_name(srv_name, MPI_INFO_NULL, port_name);
285 if (MPI_SUCCESS != ret)
286 {
287 printf("[MPI-SERVER] ERROR: server %s not found\n", dns_name);
288 continue;
289 }
290 }
291
292 // Connect with servers
293 ret = MPI_Comm_connect(port_name, MPI_INFO_NULL, 0, MPI_COMM_SELF, &server);
294 if (MPI_SUCCESS != ret)
295 {
296 printf("[MPI-SERVER] ERROR: MPI_Comm_connect fails\n");
297 continue;
298 }
299
300 buf = MPI_SERVER_FINALIZE;
301 MPI_Send(&buf, 1, MPI_INT, 0, 0, server);
302
303 // MPI_Comm_disconnect( &server ); //TODO: fail
304 }
305
306 // Close host file
307 fclose(file);
308
309 // MPI_Finalize();
310
311 return 0;
312}
