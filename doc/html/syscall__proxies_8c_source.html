97int dlsym_open(char *path, int flags)
98{
99 debug_info("dlsym_open: before open...\n");
100 debug_info("dlsym_open: Path => %s\n", path);
101 debug_info("dlsym_open: flags => %d\n", flags);
102
103 if (real_open == NULL)
104 {
105 real_open = (int (*)(char *, int, mode_t))dlsym(RTLD_NEXT, "open");
106 }
107
108 int fd = real_open((char *)path, flags, 0);
109
110 debug_info("dlsym_open: (%s,%o) return %d\n", path, flags, fd);
111
112 return fd;
113}
