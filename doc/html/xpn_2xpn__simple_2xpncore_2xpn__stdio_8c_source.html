82FILE *xpn_fopencookie(const char *path, const char *mode)
83{
84 FILE *stream;
85 int flags, *fd;
86 cookie_io_functions_t io_functions;
87
88 flags = O_RDONLY;
89 if (0 == strncmp(mode, "r+", 2))
90 {
91 flags = O_RDWR;
92 }
93 else if (0 == strncmp(mode, "w+", 2))
94 {
95 flags = O_RDWR | O_CREAT | O_TRUNC;
96 }
97 else if (0 == strncmp(mode, "a+", 2))
98 {
99 flags = O_RDWR;
100 }
101 else if (0 == strncmp(mode, "r", 1))
102 {
103 flags = O_RDONLY;
104 }
105 else if (0 == strncmp(mode, "w", 1))
106 {
107 flags = O_WRONLY | O_CREAT | O_TRUNC;
108 }
109 else if (0 == strncmp(mode, "a", 1))
110 {
111 flags = O_WRONLY;
112 }
113
114 fd = (int *)malloc(sizeof(int));
115 *fd = xpn_simple_open((char *)path, flags, 07000);
116
117 io_functions.read = xpn_reader;
118 io_functions.write = xpn_writer;
119 io_functions.seek = xpn_seeker;
120 io_functions.close = xpn_cleaner;
121
122 stream = fopencookie((void *)fd, mode, io_functions);
123
124 return stream;
125}

129FILE *xpn_simple_fopen(const char *filename, const char *mode)
130{
131 FILE *stream = NULL;
132 int fd = 0;
133 int flags = -1;
134
135 if (strstr(mode, "r+") != NULL)
136 {
137 flags = O_RDWR;
138 }
139 else if (strstr(mode, "r") != NULL)
140 {
141 flags = O_RDONLY;
142 }
143 else if (strstr(mode, "w+") != NULL)
144 {
145 flags = O_RDWR | O_CREAT | O_TRUNC;
146 }
147 else if (strstr(mode, "w") != NULL)
148 {
149 flags = O_WRONLY | O_CREAT | O_TRUNC;
150 }
151 else if (strstr(mode, "a+") != NULL)
152 {
153 flags = O_RDWR | O_CREAT | O_APPEND;
154 }
155 else if (strstr(mode, "a") != NULL)
156 {
157 flags = O_WRONLY | O_CREAT | O_APPEND;
158 }
159
160 if (flags >= 0)
161 {
162 fd = xpn_simple_open(filename, flags, 07000);
163 if (fd >= 0)
164 {
165 stream = malloc(sizeof(FILE));
166 memset(stream, 0, sizeof(FILE));
167 stream->_fileno = fd;
168 }
169 }
170
171 return stream;
172}
