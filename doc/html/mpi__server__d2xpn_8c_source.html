68void generateName(char *file, char *new_file)
69{
70 char aux[255];
71 unsigned long i;
72 long j;
73
74 for (j = 0, i = 0; i < strlen(file); i++)
75 {
76 switch (file[i])
77 {
78 case '/':
79 aux[j] = '_';
80 break;
81 case '_':
82 aux[j] = '_';
83 j++;
84 aux[j] = '_';
85 break;
86 default:
87 aux[j] = file[i];
88 break;
89 }
90 j++;
91 }
92 aux[j] = '\0';
93
94 sprintf(new_file, "%s/%s", DEFAULT_PATH, aux);
95 return;
96}
