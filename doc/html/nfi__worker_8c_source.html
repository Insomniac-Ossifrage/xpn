41void *nfi_worker_run(struct nfi_worker *args)
42{
43 struct nfi_worker *wrk;
44 ssize_t ret;
45 int is_true;
46
47 pthread_mutex_lock(&(global_mt));
48 wrk = args;
49 global_busy = 0;
50 pthread_cond_signal(&(global_cnd));
51 pthread_mutex_unlock(&(global_mt));
52
53 ret = -1;
54 is_true = 1;
55 while (is_true)
56 {
57 pthread_mutex_lock(&(wrk->mt));
58 while (!(wrk->ready))
59 {
60 pthread_cond_wait(&(wrk->cnd), &(wrk->mt));
61 }
62
63 // ret = nfi_do_operation(wrk) ;
64 ret = 1; // TMP
65
66 wrk->arg.result = ret;
67 wrk->ready = 0;
68 pthread_cond_signal(&(wrk->cnd));
69 pthread_mutex_unlock(&(wrk->mt));
70 }
71
72 return NULL;
73}

82int nfi_worker_init(struct nfi_worker *wrk, struct nfi_server *serv, int thread)
83{
84 pthread_attr_t attr;
85 int ret;
86
87 if (!global_busy)
88 {
89 /* init the global condition and mutex */
90 global_busy = 1;
91 if (!create_m)
92 {
93 create_m = 1;
94 pthread_mutex_init(&(global_mt), NULL);
95 pthread_cond_init(&(global_cnd), NULL);
96 }
97 }
98
99 /* init the nfi_worker */
100 /* pointer reference to the server */
101 wrk->server = serv;
102
103 thread = 1; // FIXME: Needed since the last changes in the threads architecture
104 wrk->thread = thread;
105
106 if (thread)
107 {
108 pthread_mutex_init(&(wrk->mt), NULL);
109 pthread_cond_init(&(wrk->cnd), NULL);
110
111 /* create the thread */
112 pthread_attr_init(&attr);
113 pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);
114 pthread_attr_setstacksize(&attr, (256 * KB));
115
116 debug_info("[NFI_WORKER] pthread_create(%lu)", (unsigned long int)pthread_self())
117
118 ret = pthread_create(&(wrk->pth), &attr, (void *(*)(void *))(nfi_worker_run), (void *)wrk);
119
120 // prevent a memory leak
121 pthread_detach(wrk->pth); // Although it doesn't seem to work
122
123 pthread_attr_destroy(&attr);
124
125 if (ret != 0)
126 {
127 perror("nfi_worker_init: Error pthread_create ");
128 }
129
130 pthread_mutex_lock(&(global_mt));
131 while (global_busy)
132 {
133 pthread_cond_wait(&(global_cnd), &(global_mt));
134 }
135 global_busy = 1;
136 pthread_mutex_unlock(&(global_mt));
137 }
138
139 return 0;
140}

175int nfi_worker_end(struct nfi_worker *wrk)
176{
177 if (wrk->thread)
178 {
179 debug_info("[NFI_WORKER] nfi_worker_end(%lu) with_threads", (unsigned long int)pthread_self())
180 pthread_mutex_destroy(&(wrk->mt));
181 pthread_cond_destroy(&(wrk->cnd));
182 }
183
184 free(wrk);
185 wrk = NULL;
186 return 0;
