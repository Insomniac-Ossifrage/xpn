70void TIME_MISC_AddTime(struct timeval *to,
71 struct timeval *tn,
72 struct timeval *sum)
73{
74 long aux;
75
76 aux = tn->tv_usec + to->tv_usec;
77 sum->tv_usec = aux % USECPSEC;
78 sum->tv_sec = (tn->tv_sec + to->tv_sec) + (aux / USECPSEC);
79}
