45ssize_t xpn_simple_read(int fd, void *buffer, size_t size)
46{
47 ssize_t res = -1;
48
49 XPN_DEBUG_BEGIN_CUSTOM("%d, %zu", fd, size)
50
51 // checks
52 if (NULL == xpn_file_table[fd])
53 {
54 XpnShowFileTable();
55 errno = EBADF;
56 return -1;
57 }
58
59 // action
60 if ((unsigned long)(size) > (unsigned long)(xpn_file_table[fd]->block_size))
61 {
62 res = xpn_pread(fd, buffer, size, xpn_file_table[fd]->offset);
63 }
64 else
65 {
66 res = xpn_sread(fd, buffer, size, xpn_file_table[fd]->offset);
67 }
68
69 XPN_DEBUG_END_CUSTOM("%d, %zu", fd, size)
70
71 return res;
72}

103ssize_t xpn_sread(int fd, const void *buffer, size_t size, off_t offset)
104{
105 off_t new_offset, l_offset;
106 size_t l_size;
107 int l_serv;
108 size_t count = 0;
109 struct nfi_server **servers;
110 int n, res;
111
112 XPN_DEBUG_BEGIN_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
113
114 if ((fd < 0) || (fd > XPN_MAX_FILE) || (xpn_file_table[fd] == NULL))
115 {
116 XpnShowFileTable();
117 errno = EBADF;
118 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
119 return -1;
120 }
121
122 if (buffer == NULL)
123 {
124 errno = EFAULT;
125 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
126 return -1;
127 }
128
129 if (size == 0)
130 {
131 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
132 return 0;
133 }
134
135 if (xpn_file_table[fd]->mode == O_WRONLY)
136 {
137 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
138 return -1;
139 }
140
141 if (xpn_file_table[fd]->type == XPN_DIR)
142 {
143 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
144 return -1;
145 }
146
147 /* params:
148 * flag operation, partition id, absolute path, file descriptor,
149 * pointer to servers */
150 servers = NULL;
151 n = XpnGetServers(op_xpn_read, xpn_file_table[fd]->part->id, NULL, fd, &servers, XPN_DATA_SERVER);
152 if (n <= 0)
153 {
154 if (servers != NULL)
155 {
156 free(servers);
157 servers = NULL;
158 }
159 return -1;
160 }
161
162 new_offset = offset;
163 count = 0;
164
165 do
166 {
167 XpnGetBlock(fd, new_offset, &l_offset, &l_serv);
168
169 l_size = xpn_file_table[fd]->block_size - (new_offset % xpn_file_table[fd]->block_size);
170
171 // If l_size > the remaining bytes to read/write, then adjust l_size
172 if ((size - count) < l_size)
173 {
174 l_size = size - count;
175 }
176
177 if (xpn_file_table[fd]->data_vfh->nfih[l_serv] == NULL)
178 {
179 res = XpnGetFh(xpn_file_table[fd]->mdata, &(xpn_file_table[fd]->data_vfh->nfih[l_serv]), servers[l_serv], xpn_file_table[fd]->path);
180 if (res < 0)
181 {
182 return -1;
183 }
184 }
185
186 res = servers[l_serv]->ops->nfi_read(servers[l_serv], xpn_file_table[fd]->data_vfh->nfih[l_serv], (char *)buffer + count, l_offset + XPN_HEADER_SIZE, l_size);
187 if (res < 0)
188 {
189 return (0 == count) ? (size_t)-1 : count;
190 }
191 count = count + res;
192 new_offset = offset + count;
193 } while ((size > count) && (res > 0));
194
195 if (servers != NULL)
196 {
197 free(servers);
198 servers = NULL;
199 }
200
201 if (count > 0)
202 {
203 xpn_file_table[fd]->offset += count;
204 }
205
206 return count;
207}

209ssize_t xpn_pread(int fd, void *buffer, size_t size, off_t offset)
210{
211 ssize_t *res_v, total;
212 int res, n, i, j, err;
213 struct nfi_server **servers;
214 struct nfi_worker_io **io;
215 int *ion, max;
216 void *new_buffer;
217
218 XPN_DEBUG_BEGIN_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
219
220 if ((fd < 0) || (fd > XPN_MAX_FILE) || (xpn_file_table[fd] == NULL))
221 {
222 XpnShowFileTable();
223 errno = EBADF;
224 res = -1;
225 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
226 return res;
227 }
228
229 if (buffer == NULL)
230 {
231 errno = EFAULT;
232 res = -1;
233 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
234 return res;
235 }
236
237 if (size == 0)
238 {
239 res = 0;
240 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
241 return res;
242 }
243
244 if (xpn_file_table[fd]->mode == O_WRONLY)
245 {
246 res = -1;
247 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
248 return res;
249 }
250
251 if (xpn_file_table[fd]->type == XPN_DIR)
252 {
253 res = -1;
254 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
255 return res;
256 }
257
258 /* params:
259 * flag operation, partition id, absolute path, file descriptor,
260 * pointer to servers */
261 servers = NULL;
262 n = XpnGetServers(op_xpn_read, xpn_file_table[fd]->part->id, NULL, fd, &servers, XPN_DATA_SERVER);
263 if (n <= 0)
264 {
265 if (servers != NULL)
266 {
267 free(servers);
268 servers = NULL;
269 }
270
271 res = -1;
272 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
273 return res;
274 }
275
276 io = (struct nfi_worker_io **)malloc(sizeof(struct nfi_worker_io *) * n);
277 if (io == NULL)
278 {
279 if (servers != NULL)
280 {
281 free(servers);
282 servers = NULL;
283 }
284
285 res = -1;
286 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
287 return res;
288 }
289
290 ion = (int *)malloc(sizeof(int) * n);
291 if (ion == NULL)
292 {
293 if (servers != NULL)
294 {
295 free(servers);
296 servers = NULL;
297 }
298 free(io);
299
300 res = -1;
301 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
302 return res;
303 }
304
305 bzero(ion, sizeof(int) * n);
306
307 // calculate the maximum number of operations
308 max = (size / xpn_file_table[fd]->block_size) + 1;
309 if (size % xpn_file_table[fd]->block_size != 0)
310 {
311 max++;
312 }
313
314 // create
315 for (i = 0; i < n; i++)
316 {
317 io[i] = (struct nfi_worker_io *)malloc(sizeof(struct nfi_worker_io) * max);
318 if (io[i] == NULL)
319 {
320 if (servers != NULL)
321 {
322 free(servers);
323 servers = NULL;
324 }
325
326 for (j = 0; j < i; j++)
327 {
328 free(io[j]);
329 }
330 free(io);
331 free(ion);
332
333 res = -1;
334 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
335 return res;
336 }
337 }
338
339 for (i = 0; i < n; i++)
340 {
341 io[i][0].offset = 0;
342 io[i][0].size = 0;
343 }
344
345 // Calculate which blocks to read from each server
346 new_buffer = XpnReadBlocks(fd, buffer, size, offset, &io, &ion, n);
347 if (new_buffer == NULL)
348 {
349 if (servers != NULL)
350 {
351 free(servers);
352 servers = NULL;
353 }
354
355 for (i = 0; i < n; i++)
356 {
357 free(io[i]);
358 }
359
360 free(io);
361 free(ion);
362 return -1;
363 }
364
365 // TODO: Expand is not reentrant. Using only one 'wrk' (struct nfi_worker) inside servers[i] (struct nfi_server) does not allow two nfi operations at the same time
366 // pthread_mutex_lock(&(global_mt)); // It is better to do this in clients when necessary
367
368 // operation
369 for (j = 0; j < n; j++)
370 {
371 i = XpnGetMetadataPos(xpn_file_table[fd]->mdata, j);
372
373 if (ion[i] != 0)
374 {
375 res = XpnGetFh(xpn_file_table[fd]->mdata, &(xpn_file_table[fd]->data_vfh->nfih[i]), servers[i], xpn_file_table[fd]->path);
376 if (res < 0)
377 {
378 if (servers != NULL)
379 {
380 free(servers);
381 servers = NULL;
382 }
383
384 for (i = 0; i < n; i++)
385 {
386 free(io[i]);
387 }
388
389 free(io);
390 free(ion);
391
392 res = -1;
393 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
394 return res;
395 }
396
397 // Worker
398 servers[i]->wrk->thread = servers[i]->xpn_thread;
399 nfi_worker_do_read(servers[i]->wrk, xpn_file_table[fd]->data_vfh->nfih[i], io[i], ion[i]);
400 }
401 }
402
403 res_v = (ssize_t *)malloc(sizeof(ssize_t) * n);
404 if (res_v == NULL)
405 {
406 if (servers != NULL)
407 {
408 free(servers);
409 servers = NULL;
410 }
411
412 for (j = 0; j < n; j++)
413 {
414 free(io[j]);
415 }
416
417 free(io);
418 free(ion);
419
420 return -1;
421 }
422
423 bzero(res_v, sizeof(ssize_t) * n);
424
425 err = 0;
426 for (i = 0; i < n; i++)
427 {
428 if (ion[i] != 0)
429 {
430 res_v[i] = nfiworker_wait(servers[i]->wrk);
431 if (res_v[i] < 0)
432 {
433 err = 1;
434 }
435 }
436 }
437
438 // pthread_mutex_unlock(&(global_mt));
439
440 XpnReadBlocksFinish(fd, buffer, size, offset, &io, &ion, n, new_buffer);
441
442 total = -1;
443 if (!err)
444 {
445 total = XpnReadGetTotalBytes(fd, res_v, n);
446
447 if (total > 0)
448 {
449 xpn_file_table[fd]->offset += total;
450 }
451 }
452
453 if (servers != NULL)
454 {
455 free(servers);
456 servers = NULL;
457 }
458
459 for (j = 0; j < n; j++)
460 {
461 free(io[j]);
462 }
463
464 free(io);
465 free(ion);
466
467 res = total;
468 XPN_DEBUG_END_CUSTOM("%d, %zu, %lld", fd, size, (long long int)offset)
469
470 return res;
471}

833off_t xpn_simple_lseek(int fd, off_t offset, int flag)
834{
835 struct stat st;
836
837 switch (flag)
838 {
839 case SEEK_SET:
840 if (offset < 0)
841 {
842 errno = EINVAL;
843 return (off_t)-1;
844 }
845 else
846 {
847 xpn_file_table[fd]->offset = offset;
848 }
849 break;
850
