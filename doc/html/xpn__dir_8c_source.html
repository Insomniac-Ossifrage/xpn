243 if (ret < 0)
244 {

249 pd = XpnGetPartition(abs_path); /* return partition's id */

267 {
268 XpnGetURLServer(servers[i], abs_path, url_serv);
269 // Worker
270 nfi_worker_do_rmdir(servers[i]->wrk, url_serv);
271 }
272
273 // Wait
274 err = 0;
275 for (i = 0; i < n; i++)
276 {
277 ret = nfiworker_wait(servers[i]->wrk);
278 // Error checking
279 if ((ret < 0) && (!err))
280 {
281 err = 1;
282 }
283 }
284
285 // Error checking
286 if (err)
287 {
288 xpn_err(XPNERR_REMOVE);
289 free(servers);
290 return -1;
291 }
292
293 free(servers);
294 return 0;
