77int xpn_open(const char *path, int flags, ...)
78{
79 va_list ap;
80 int ret = -1;
81 mode_t mode = 0;
82
83 // get mode
84 if ((flags & O_CREAT) > 0)
85 {
86 va_start(ap, flags);
87 mode = va_arg(ap, mode_t);
88 va_end(ap);
89 }
90
91 // call simple_open
92 ret = xpn_simple_open(path, flags, mode);
93
94 // return ret
95 return ret;
96}
