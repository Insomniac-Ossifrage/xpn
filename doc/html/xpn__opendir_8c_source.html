139int xpn_simple_closedir(DIR *dirp)
140{
141 int i;
142
143 if ((NULL == dirp) || (dirp->fd < 0) || (dirp->fd > XPN_MAX_FILE - 1))
144 {
145 // set errno
146 return -1;
147 }
148
149 if (xpn_file_table[dirp->fd] == NULL)
150 {
151 // xpn_err
152 return -1;
153 }
154
155 xpn_file_table[dirp->fd]->links--;
156
157 if (xpn_file_table[dirp->fd]->links == 0)
158 {
159 for (i = 0; i < xpn_file_table[dirp->fd]->data_vfh->n_nfih; i++)
160 {
161 if (xpn_file_table[dirp->fd]->data_vfh->nfih[i] != NULL)
162 {
163 if (xpn_file_table[dirp->fd]->data_vfh->nfih[i]->priv_fh != NULL)
164 {
165 xpn_file_table[dirp->fd]->data_vfh->nfih[i]->server->ops->nfi_closedir(xpn_file_table[dirp->fd]->data_vfh->nfih[i]->server, xpn_file_table[dirp->fd]->data_vfh->nfih[i]);
166 }
167
168 free(xpn_file_table[dirp->fd]->data_vfh->nfih[i]);
169 }
170 }
171
172 free(xpn_file_table[dirp->fd]->data_vfh->nfih);
173 free(xpn_file_table[dirp->fd]->data_vfh);
174
175 free(xpn_file_table[dirp->fd]->mdata->policy);
176 free(xpn_file_table[dirp->fd]->mdata);
177
178 free(xpn_file_table[dirp->fd]);
179 xpn_file_table[dirp->fd] = NULL;
180 }
181
182 free(dirp->path);
183 free(dirp);
184
185 // set errno
186 xpn_err(XPN_OK);
187 return 0;
188 // return -1;
189}
