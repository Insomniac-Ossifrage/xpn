95int tcp_server_params_get(tcp_server_param_st *params, int argc, char *argv[])
96{
97 DEBUG_BEGIN();
98
99 // set default values
100 params->argc = argc;
101 params->argv = argv;
102 params->size = 0;
103 params->rank = 0;
104 params->thread_mode = TH_POOL;
105 strcpy(params->port_name, "");
106 strcpy(params->srv_name, "");
107 strcpy(params->dirbase, TCP_SERVER_DIRBASE_DEFAULT);
108 strcpy(params->dns_file, TCP_SERVER_DNS_FILE_DEFAULT);
109
110 gethostname(params->name, TCP_MAX_PORT_NAME);
111 sprintf(params->port, "%d", TCP_SERVER_PORT_DEFAULT);
112 params->IOsize = TCP_SERVER_IOSIZE_DEFAULT;
113
114 params->mosquitto_mode = 0;
115 params->mosquitto_qos = 0;
116
117 // update user requests
118 for (int i = 0; i < argc; i++)
119 {
120 switch (argv[i][0])
121 {
122 case '-':
123 switch (argv[i][1])
124 {
125 case 'p':
126 strcpy(params->port, argv[i + 1]);
127 i++;
128 break;
129 case 'n':
130 if ((strlen(argv[i]) == 2))
131 {
132 strcpy(params->name, argv[i + 1]);
133 i++;
134 }
135 if ((strlen(argv[i]) == 3) && (argv[i][2] == 's'))
136 {
137 strcpy(params->dns_file, argv[i + 1]);
138 i++;
139 }
140 if ((strlen(argv[i]) == 4) && (argv[i][2] == 't') && (argv[i][3] == 'p'))
141 {
142 params->thread_mode = TH_OP;
143 i++;
144 }
145 break;
146 case 'i':
147 if ((strlen(argv[i]) == 3) && (argv[i][2] == 'o'))
148 {
149 params->IOsize = atoi(argv[i + 1]);
150 i++;
151 }
152 break;
153 case 'f':
154 strcpy(params->shutdown_file, argv[i + 1]);
155 i++;
156 break;
157 case 'd':
158 strcpy(params->dirbase, argv[i + 1]);
159 i++;
160 break;
161 case 't':
162 if (isdigit(argv[i + 1][0]))
163 {
164 int thread_mode_aux = atoi(argv[i + 1]);
165
166 if (thread_mode_aux >= TH_NOT && thread_mode_aux <= TH_OP)
167 {
168 params->thread_mode = thread_mode_aux;
169 }
170 else
171 {
172 printf("ERROR: unknown option %s\n", argv[i + 1]);
173 }
174 }
175 else
176 {
177 if (strcmp("without", argv[i + 1]) == 0)
178 {
179 params->thread_mode = TH_NOT;
180 }
181 else if (strcmp("pool", argv[i + 1]) == 0)
182 {
183 params->thread_mode = TH_POOL;
184 }
185 else if (strcmp("on_demand", argv[i + 1]) == 0)
186 {
187 params->thread_mode = TH_OP;
188 }
189 else
190 {
191 printf("ERROR: unknown option %s\n", argv[i + 1]);
192 }
193 }
194 i++;
195 break;
196 case 'm':
197 params->mosquitto_mode = 1;
198
199 if (isdigit(argv[i + 1][0]))
200 {
201 int qos_mode_mqtt = atoi(argv[i + 1]);
202
203 if (qos_mode_mqtt < 0 || qos_mode_mqtt > 2)
204 {
205 printf("ERROR: unknown QoS value for MQTT. Default value 0 selected\n");
206 }
207 else
208 {
209 params->mosquitto_qos = qos_mode_mqtt;
210 }
211 }
212 else
213 {
214 printf("ERROR: unknown QoS value for MQTT. Default value 0 selected\n");
215 }
216
217 i++;
218 break;
219 case 'h':
220 return -1;
221
222 default:
223 break;
224 }
225 break;
226
227 default:
228 break;
229 }
230 }
231
232 // return OK
233 DEBUG_END();
234 return 1;
235}
