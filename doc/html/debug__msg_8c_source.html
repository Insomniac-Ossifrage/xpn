53int debug_msg_printf(int src_type, char *src_fname, long src_line, FILE *fd, char *msg_fmt, ...)
54{
55 va_list valist;
56 int ret;
57
58 va_start(valist, msg_fmt);
59 switch (src_type)
60 {
61 case 3:
62 fprintf(fd, "[%s:%4ld] [INFO] ", src_fname, src_line);
63 ret = vfprintf(fd, msg_fmt, valist);
64 break;
65
66 case 2:
67 fprintf(fd, "[%s:%4ld] [WARN] ", src_fname, src_line);
68 ret = vfprintf(fd, msg_fmt, valist);
69 break;
70
71 case 1:
72 fprintf(fd, "[%s:%4ld] [ERROR] ", src_fname, src_line);
73 ret = vfprintf(fd, msg_fmt, valist);
74 break;
75
76 default:
77 ret = vfprintf(fd, msg_fmt, valist);
78 break;
79 }
80 va_end(valist);
81
82 // fflush(fd) ;
83
84 return ret;
85}
