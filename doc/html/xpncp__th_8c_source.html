52void print_bar()
53{
54 int bar_expected_length;
55
56 memcpy(&t_old_transfer, &t_end_transfer, sizeof(struct timeval));
57
58 gettimeofday(&t_end_transfer, NULL);
59
60 transfer_t = (t_end_transfer.tv_sec - t_old_transfer.tv_sec) + (double)(t_end_transfer.tv_usec - t_old_transfer.tv_usec) / 1000000;
61 transfer_bw = (sum - sum_old) / transfer_t;
62 if (st.st_size > 0)
63 {
64 bar_expected_length = sum / ((float)(st.st_size) / BAR_LENGTH);
65 while (bar_expected_length > bar_length)
66 {
67 bar[bar_length] = '=';
68 bar_length++;
69 }
70 }
71
72 if (!silent)
73 {
74 printf("%c[%s] %zd %.3f MB/s", (char)13, bar, sum, transfer_bw / MB);
75 fflush(NULL);
76 }
77}

79void *progression_bar(__attribute__((unused)) void *arg)
80{
81 int i;
82 int is_true = 1;
83
84 for (i = 0; i < BAR_LENGTH; i++)
85 bar[i] = ' ';
86 bar[BAR_LENGTH] = '\0';
87
88 memcpy(&t_end_transfer, &t_ini_transfer, sizeof(struct timeval));
89
90 bar_length = 0;
91 sum_old = 0;
92
93 if (!very_silent)
94 {
95 printf("\n");
96 }
97
98 while (is_true)
99 {
100 print_bar();
101 sum_old = sum;
102 sleep(1);
103 }
104
105 pthread_exit(0);
106 return NULL;
107}

129void *write_func(void *func_args)
130{
131 struct write_args *args = (struct write_args *)func_args;
132 struct timeval t_ini_trans, t_end_trans;
133 ssize_t nw;
134
135 while (sum < st.st_size)
136 {
137 pthread_mutex_lock(&mutex);
138 while (next >= last)
139 {
140 // fprintf(stderr, "write_func: cond not ready next %zd last %zd\n", next, last);
141 pthread_cond_wait(&cond, &mutex);
142 }
143 pthread_mutex_unlock(&mutex);
144
145 if (!very_silent)
146 gettimeofday(&t_ini_trans, NULL);
147 if (args->xpndest)
148 {
149 // Expand is not reentrant
150 if (args->xpnsource)
151 pthread_mutex_lock(&xpn_mutex);
152 nw = xpn_write(args->fdd, array_bufs[next]->buffer, array_bufs[next]->buffer_length);
153 if (args->xpnsource)
154 pthread_mutex_unlock(&xpn_mutex);
155 }
156 else
157 {
158 nw = write(args->fdd, array_bufs[next]->buffer, array_bufs[next]->buffer_length);
159 fsync(args->fdd); // Improves performance
160 }
161
162 if (!very_silent)
163 {
164 gettimeofday(&t_end_trans, NULL);
165 *(args->out_transfer_t) += (t_end_trans.tv_sec - t_ini_trans.tv_sec) + (double)(t_end_trans.tv_usec - t_ini_trans.tv_usec) / 1000000;
166 }
167
168 if (debug)
169 fprintf(stderr, "write(%zd) written block %zd (%p) of %zd bytes %u %u %u %u ... %u %u %u %u\n", array_bufs[next]->buffer_length, next, array_bufs[next]->buffer, nw, (unsigned char)array_bufs[next]->buffer[0], (unsigned char)array_bufs[next]->buffer[1], (unsigned char)array_bufs[next]->buffer[2], (unsigned char)array_bufs[next]->buffer[3], (unsigned char)array_bufs[next]->buffer[nw - 4], (unsigned char)array_bufs[next]->buffer[nw - 3], (unsigned char)array_bufs[next]->buffer[nw - 2], (unsigned char)array_bufs[next]->buffer[nw - 1]);
170
171 sum += nw;
172
173 free(array_bufs[next]->buffer);
174 array_bufs[next]->buffer = NULL;
175 free(array_bufs[next]);
176 array_bufs[next] = NULL;
177
178 next++;
179
180 if (debug)
181 printf("Written a block of %zu bytes\n", nw);
182 }
183
184 pthread_exit(0);
185 return NULL;
186}

252/* Parse a single option. */
253static error_t parse_opt(int key, char *arg, struct argp_state *state)
254{
255 /* Get the input argument from argp_parse, which we
256 know is a pointer to our arguments structure. */
257 struct arguments *arguments = state->input;
258 char *suffix;
259 int units = 0 /*, i*/;
260
261 switch (key)
262 {
263 case 'b':
264 units = 1;
265 arguments->buffer_size = strtoul(arg, &suffix, 10);
266 if (strlen(suffix) > 0)
267 {
268 if (strlen(suffix) > 1)
269 argp_usage(state);
270 switch (*suffix)
271 {
272 case 'k':
273 case 'K':
274 units = 1024;
275 break;
276 case 'm':
277 case 'M':
278 units = 1024 * 1024;
279 break;
280 case 'g':
281 case 'G':
282 units = 1024 * 1024 * 1024;
283 break;
284 default:
285 fprintf(stderr, "Unknown suffix '%s' when parsing buffer_size '%s'\n", suffix, arg);
286 argp_state_help(state, stdout, ARGP_HELP_STD_HELP);
287 argp_usage(state);
288 }
289 if (arguments->buffer_size * (long long)units < SSIZE_MAX)
290 arguments->buffer_size *= units;
291 else
292 arguments->buffer_size = SSIZE_MAX;
293 }
294 break;
295 case 'i':
296 units = 1;
297 arguments->st.st_size = strtoll(arg, &suffix, 10);
298 arguments->do_stat = 0;
299 if (strlen(suffix) > 0)
300 {
301 if (strlen(suffix) > 1)
302 argp_usage(state);
303 switch (*suffix)
304 {
305 case 'k':
306 case 'K':
307 units = 1024;
308 break;
309 case 'm':
310 case 'M':
311 units = 1024 * 1024;
312 break;
313 case 'g':
314 case 'G':
315 units = 1024 * 1024 * 1024;
316 break;
317 default:
318 fprintf(stderr, "Unknown suffix '%s' when parsing file_size '%s'\n", suffix, arg);
319 argp_state_help(state, stdout, ARGP_HELP_STD_HELP);
320 argp_usage(state);
321 }
322 arguments->st.st_size *= units;
323 }
324 break;
325 case 's':
326 arguments->silent = 1;
327 break;
328 case 'S':
329 arguments->silent = 1;
330 arguments->very_silent = 1;
331 break;
332 case 'd':
333 arguments->debug = 1;
334 break;
335 case ARGP_KEY_ARG:
336 return ARGP_ERR_UNKNOWN;
337 case ARGP_KEY_ARGS:
338 arguments->source = state->argv[state->next];
339 // TODO: Handle several input files
340 // for (i = state->next ; i < state->argc ; i++ )
341 // printf("arg=%s\n", state->argv[i]);
342 arguments->dest = state->argv[state->argc - 1];
343 break;
344 case ARGP_KEY_END:
345 // if (state->arg_num < 2)
346 if (state->arg_num != 2)
347 /* Not enough arguments. */
348 argp_usage(state);
349 break;
350 default:
351 return ARGP_ERR_UNKNOWN;
352 }
353 return 0;
354}

360{

626 }
