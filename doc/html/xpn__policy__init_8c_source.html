65struct conf_connect_st *XpnPartitionOpen(void)
66{
67 static struct conf_connect_st desc;
68 char conf[PATH_MAX];
69
70 desc.type = CONF_DEFAULT;
71 switch (desc.type)
72 {
73 case CONF_FILE:
74 if (param_get(XPN_CONF) != NULL)
75 {
76 strcpy(conf, param_get(XPN_CONF));
77 }
78 else
79 {
80 strcpy(conf, XPN_CONF_DEFAULT);
81 }
82 desc.connect_u.f = fopen(conf, "r");
83 if (desc.connect_u.f == NULL)
84 {
85 fprintf(stderr, "xpn_init: Can't open %s\n", conf);
86 return NULL;
87 }
88 break;
89
90 case CONF_MXML:
91#ifdef ENABLE_MXML
92 if (param_get(XPN_CONF) != NULL)
93 {
94 strcpy(desc.connect_u.xml.conf_n, param_get(XPN_CONF));
95 }
96 else
97 {
98 strcpy(desc.connect_u.xml.conf_n, XPN_CONF_DEFAULT);
99 }
100
101 if (param_get(XPN_PROFILE) != NULL)
102 {
103 strcpy(desc.connect_u.xml.profile_n, param_get(XPN_PROFILE));
104 }
105 else
106 {
107 strcpy(desc.connect_u.xml.profile_n, XPN_PROFILE_DEFAULT);
108 }
109
110 XPN_DEBUG("Using XPN_CONF=%s", desc.connect_u.xml.conf_n);
111 XPN_DEBUG("Using XPN_PROFILE=%s", desc.connect_u.xml.profile_n);
112
113 desc.connect_u.xml.conf_f = fopen(desc.connect_u.xml.conf_n, "r");
114 if (desc.connect_u.xml.conf_f == NULL)
115 {
116 fprintf(stderr, "xpn_init: Can't open %s\n", desc.connect_u.xml.conf_n);
117 return NULL;
118 }
119
120 desc.connect_u.xml.conf_tree = mxmlLoadFile(NULL, desc.connect_u.xml.conf_f, MXML_NO_CALLBACK);
121 if (desc.connect_u.xml.conf_tree == NULL)
122 {
123 fclose(desc.connect_u.xml.conf_f);
124 fprintf(stderr, "(1)xpn_init: %s is empty.\n", desc.connect_u.xml.conf_n);
125 return NULL;
126 }
127
128 desc.connect_u.xml.conf_partition_node = desc.connect_u.xml.conf_tree;
129 desc.connect_u.xml.profile_f = fopen(desc.connect_u.xml.profile_n, "r");
130
131#endif
132 break;
133 }
134
135 return &desc;
136}

213int XpnGetInfoPartition(struct conf_connect_st *fconf, struct xpn_partition *part)
214{
215#ifdef ENABLE_MXML
216 char *value = NULL;
217 char *value_th = NULL;
218 mxml_node_t *node;
219#endif
220 char type[20]; // TODO: 20??
221
222 memset(type, 0, 20);
223
224 switch (fconf->type)
225 {
226 case CONF_FILE:
227 if (fscanf(fconf->connect_u.f, "%zu %d %s", &(part->block_size), &(part->data_nserv), type) == EOF)
228 {
229 return -1;
230 }
231 part->block_size = part->block_size * KB;
232
233 if ((strcmp(type, "RAID0") == 0) || (strcmp(type, "NORMAL") == 0))
234 { // FIXME: previously was 'value', now changed to 'type' but not tested
235 part->type = POLICY_RAID0;
236 }
237 else if (strcmp(type, "RAID1") == 0)
238 {
239 part->type = POLICY_RAID1;
240 }
241 else
242 {
243 part->type = -1;
244 }
245 break;
246
247 case CONF_MXML:
248#ifdef ENABLE_MXML
249 if (fconf->connect_u.xml.profile_f != NULL)
250 {
251 fconf->connect_u.xml.profile_tree = mxmlLoadFile(NULL, fconf->connect_u.xml.profile_f, MXML_NO_CALLBACK);
252 // if (fconf->connect_u.xml.profile_tree == NULL){
253 // fclose(fconf->connect_u.xml.profile_f);
254 // fclose(fconf->connect_u.xml.conf_f);
255 // fprintf(stderr,"(2)xpn_init: %s is empty.\n", fconf->connect_u.xml.profile_n);
256 // return 0;
257 // }
258
259 // if (fconf->connect_u.xml.profile_tree != NULL){
260 fconf->connect_u.xml.profile_partition_node = fconf->connect_u.xml.profile_tree;
261 //}
262 }
263
264 value = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_partition_node, XML_TAG_ATTR_BLOCKSIZE);
265
266 if (value == NULL)
267 {
268 value = XML_DEFAULT_ATTR_BLOCKSIZE;
269 }
270
271 part->block_size = getSizeFactor(value);
272
273 value = NULL;
274 value = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_partition_node, XML_TAG_ATTR_TYPE);
275 if (value == NULL)
276 {
277 value = XML_DEFAULT_ATTR_TYPE;
278 }
279
280 if ((strcmp(value, "RAID0") == 0) || (strcmp(value, "NORMAL") == 0))
281 {
282 part->type = POLICY_RAID0;
283 }
284 else if (strcmp(value, "RAID1") == 0)
285 {
286 part->type = POLICY_RAID1;
287 }
288 else
289 {
290 part->type = -1;
291 }
292
293 /* THREADS */
294 value = NULL;
295 value = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_partition_node, XML_TAG_ATTR_THREADS);
296
297 debug_info("[XPN]XML_TAG_ATTR_THREADS = %s\n", value ? value : "NULL");
298
299 if (value == NULL)
300 {
301 value = XML_DEFAULT_ATTR_THREADS;
302 }
303
304 switch (value[0])
305 {
306 case 'Y':
307 case 'y':
308 if (fconf->connect_u.xml.conf_partition_node != NULL)
309 {
310 value_th = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_partition_node, XML_TAG_ATTR_TH_THRESHOLD); // This is property 'th_limit'
311 // value_th = value;
312 debug_info("[XPN]XML_TAG_ATTR_TH_THRESHOLD = %s\n", value_th);
313 }
314
315 break;
316
317 default:
318 debug_info("[XPN]part->name = %s\n", part->name);
319 part->size_threads = -1;
320 debug_info("[XPN]part->size_threads: %d\n", part->size_threads);
321 break;
322 }
323
324 /*N DATASERVERS*/
325 part->data_nserv = 0;
326 node = fconf->connect_u.xml.conf_partition_node;
327 if (node != NULL)
328 {
329 for (node = mxmlFindElement(node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_DATANODE, NULL, NULL, MXML_DESCEND); node != NULL; node = mxmlFindElement(node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_DATANODE, NULL, NULL, MXML_DESCEND))
330 {
331 part->data_nserv++;
332 }
333 }
334
335 if (!part->data_nserv)
336 {
337 fprintf(stderr, "(3)xpn_init: %s has no Data Nodes inside\n", fconf->connect_u.xml.conf_n);
338 return -1;
339 }
340
341 /*N METASERVERS*/
342 part->meta_nserv = 0;
343 if (part->data_nserv > 1)
344 {
345 node = fconf->connect_u.xml.conf_partition_node;
346 if (node != NULL)
347 {
348 for (node = mxmlFindElement(node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_METADATANODE, NULL, NULL, MXML_DESCEND); node != NULL; node = mxmlFindElement(node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_METADATANODE, NULL, NULL, MXML_DESCEND))
349 {
350 part->meta_nserv++;
351 }
352 }
353 }
354
355 /* THREADS THRESHOLD */
356 debug_info("[XPN]part->data_nserv * part->block_size: %d * %d\n", part->data_nserv, part->block_size);
357
358 if (value_th == NULL)
359 {
360 part->size_threads = part->data_nserv * part->block_size; // if th_limit is not set, then th_limit = num_servers * block_size
361 }
362 else
363 {
364 part->size_threads = getSizeFactor(value_th); // else th_limit = as defined in conf file
365 }
366
367 debug_info("[XPN]part->size_threads: %d\n", part->size_threads);
368
369 return 1;
370#endif
371
372 default:
373 return -1;
374 }
375
376 return 1;
377}

399int XpnGetServer(struct conf_connect_st *fconf, __attribute__((__unused__)) struct xpn_partition *part, struct nfi_server *serv, int type)
400{
401 int ret;
402 char prt[PROTOCOL_MAXLEN];
403 char *url;
404
405 url = serv->url;
406 switch (fconf->type)
407 {
408 case CONF_FILE:
409 if ((ret = fscanf(fconf->connect_u.f, "%s", url)) == EOF)
410 {
411 return 0;
412 }
413 break;
414
415 case CONF_MXML:
416#ifdef ENABLE_MXML
417 switch (type)
418 {
419 case XPN_DATA_SERVER:
420 fconf->connect_u.xml.conf_data_node = mxmlFindElement(fconf->connect_u.xml.conf_data_node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_DATANODE, NULL, NULL, MXML_DESCEND);
421 if (fconf->connect_u.xml.conf_data_node == NULL)
422 {
423 return 0;
424 }
425
426 url = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_data_node, XML_TAG_ATTR_URL);
427 break;
428
429 case XPN_META_SERVER:
430 fconf->connect_u.xml.conf_meta_node = mxmlFindElement(fconf->connect_u.xml.conf_meta_node, fconf->connect_u.xml.conf_partition_node, XML_TAG_ELEM_METADATANODE, NULL, NULL, MXML_DESCEND);
431 if (fconf->connect_u.xml.conf_meta_node == NULL)
432 {
433 return 0;
434 }
435
436 url = (char *)mxmlElementGetAttr(fconf->connect_u.xml.conf_meta_node, XML_TAG_ATTR_URL);
437 break;
438 }
439
440 if (url == NULL)
441 {
442 return 0;
443 }
444#endif
445 break;
446
447 default:
448 fprintf(stderr, "[XPN] Conf type %d not recognized\n", fconf->type);
449 break;
450 }
451
452 XPN_DEBUG("url=%s", url);
453
454 ret = ParseURL(url, prt, NULL, NULL, NULL, NULL, NULL);
455 if (ret < 0)
456 {
457 xpn_err(XPNERR_INVALURL);
458 return -1;
459 }
460
461 serv->block_size = part->block_size; // Reference of the partition blocksize
462
463 // crear conexion
464 if (strcmp(prt, "file") == 0)
465 {
466 // printf("[XPN]nfi_local_init: %s\n",url);
467 ret = nfi_local_init(url, serv, NULL);
468 if (ret < 0)
469 {
470 xpn_err(XPNERR_INITSERV);
471 return -1;
472 }
473 }
474
475#ifdef ENABLE_NFS
476 else if ((strcmp(prt, "nfs") == 0) || (strcmp(prt, "nfs2") == 0))
477 {
478 // printf("[XPN]nfi_nfs_init: %s\n",url);
479 ret = nfi_nfs_init(url, serv, NULL);
480 if (ret < 0)
481 {
482 xpn_err(XPNERR_INITSERV);
483 return -1;
484 }
485 }
486#endif
487
488#ifdef ENABLE_NFS3
489 else if (strcmp(prt, "nfs3") == 0)
490 {
491 // printf("[XPN]nfi_nfs3_init: %s\n",url);
492 ret = nfi_nfs3_init(url, serv, NULL);
493 if (ret < 0)
494 {
495 xpn_err(XPNERR_INITSERV);
496 return -1;
497 }
498 }
499#endif
500
501#ifdef ENABLE_TCP_SERVER
502 else if (strcmp(prt, "tcp_server") == 0)
503 {
504 // printf("[XPN]nfi_tcp_server_init: %s\n",url);
505 ret = nfi_tcp_server_init(url, serv, NULL);
506 if (ret < 0)
507 {
508 xpn_err(XPNERR_INITSERV);
509 return -1;
510 }
511 }
512#endif
513
514#ifdef ENABLE_MPI_SERVER
515 else if (strcmp(prt, "mpi_server") == 0)
516 {
517 // printf("[XPN]nfi_mpi_server_init: %s\n",url);
518 ret = nfi_mpi_server_init(url, serv, NULL);
519 if (ret < 0)
520 {
521 xpn_err(XPNERR_INITSERV);
522 return -1;
523 }
524 }
525#endif
526
527#ifdef ENABLE_HTTP
528 else if ((strcmp(prt, "http") == 0) || (strcmp(prt, "webdav") == 0))
529 {
530 ret = nfi_http_init(url, serv, NULL);
531 if (ret < 0)
532 {
533 xpn_err(XPNERR_INITSERV);
534 return -1;
535 }
536 }
537#endif
538
539 else
540 {
541 printf("[XPN] Protocol '%s' not recognized\n", prt);
542 xpn_err(XPNERR_INVALURL);
543 return -1;
544 }
545
546 // Default Value
547 return 1;
548}

550int XpnGetPartition(char *path) /* return partition's id */
551{
552 int i;
553 char part[PATH_MAX];
554
555 getNamePart(part, path);
556
557 i = 0;
558 while ((i < XPN_MAX_PART) && (strcmp(part, xpn_parttable[i].name) != 0))
559 {
560 i++;
561 }
562
563 if (i == XPN_MAX_PART)
564 {
565 fprintf(stderr, "Error: Partition '%s' does not exist in the conf file.\n", part);
566 return -1;
567 }
568
569 return xpn_parttable[i].id;
570}
