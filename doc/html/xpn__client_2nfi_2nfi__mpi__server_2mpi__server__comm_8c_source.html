40int mpiClient_comm_init(mpiClient_param_st *params)
41{
42 int ret, provided, claimed;
43 int flag = 0;
44
45 debug_info("[CLI-COMM] begin mpiClient_comm_init(...)\n");
46
47 // MPI_Init
48 MPI_Initialized(&flag);
49
50 if (!flag)
51 {
52 // TODO: server->argc, server->argv from upper layers?
53
54 // Threads disable
55 if (!params->xpn_thread)
56 {
57 ret = MPI_Init(NULL, NULL);
58 if (MPI_SUCCESS != ret)
59 {
60 debug_error("Server[%d]: MPI_Init fails :-(", -1);
61 return -1;
62 }
63 }
64 else
65 {
66 ret = MPI_Init_thread(NULL, NULL, MPI_THREAD_MULTIPLE, &provided);
67 if (MPI_SUCCESS != ret)
68 {
69 debug_error("Server[%d]: MPI_Init_thread fails :-(", -1);
70 return -1;
71 }
72
73 MPI_Query_thread(&claimed);
74 if (claimed != MPI_THREAD_MULTIPLE)
75 {
76 printf("MPI_Init_thread: your MPI implementation seem not supporting thereads\n");
77 }
78 }
79 }
80
81 // params->rank = comm_rank()
82 ret = MPI_Comm_rank(MPI_COMM_WORLD, &(params->rank));
83 if (MPI_SUCCESS != ret)
84 {
85 debug_error("Server[%d]: MPI_Comm_rank fails :-(", params->rank);
86 return -1;
87 }
88
89 // params->size = comm_size()
90 ret = MPI_Comm_size(MPI_COMM_WORLD, &(params->size));
91 if (MPI_SUCCESS != ret)
92 {
93 debug_error("Server[%d]: MPI_Comm_size fails :-(", params->size);
94 return -1;
95 }
96
97 // set is_mpi_server as the used protocol
98 setenv("XPN_IS_MPI_SERVER", "1", 1);
99
100 debug_info("[CLI-COMM] server %d available at %s\n", params->rank, params->port_name);
101
102 debug_info("[CLI-COMM] end mpiClient_comm_init(...)\n");
103
104 // Return OK
105 return 1;
106}

144int mpiClient_comm_connect(mpiClient_param_st *params)
145{
146 int ret;
147
148 debug_info("[CLI-COMM] begin mpiClient_comm_connect(...)\n");
149
150 int version_len;
151 char version[MPI_MAX_LIBRARY_VERSION_STRING];
152 MPI_Get_library_version(version, &version_len);
153
154 if (strncasecmp(version, "Open MPI", strlen("Open MPI")) != 0)
155 {
156 // Lookup port name
157 int lookup_retries = 0;
158 char aux_srv_ip[1024];
159 do
160 {
161 ret = ns_lookup("mpi_server", params->srv_name, aux_srv_ip, params->port_name);
162 if (ret < 0)
163 {
164 if (lookup_retries == 0)
165 {
166 char cli_name[HOST_NAME_MAX];
167 gethostname(cli_name, HOST_NAME_MAX);
168 printf("----------------------------------------------------------------\n");
169 printf("XPN Client %s : Waiting for servers being up and runing...\n", cli_name);
170 printf("----------------------------------------------------------------\n\n");
171 }
172 lookup_retries++;
173 sleep(2);
174 }
175 } while ((ret < 0) && (lookup_retries < 150));
176
177 if (ret < 0)
178 {
179 debug_error("ERROR: DNS Lookup %s Port %s\n", params->srv_name, params->port_name);
180 return -1;
181 }
182 }
183 else
184 {
185 // Lookup port name on nameserver
186 ret = MPI_Lookup_name(params->srv_name, MPI_INFO_NULL, params->port_name);
187 if (MPI_SUCCESS != ret)
188 {
189 debug_error("Server[%d]: MPI_Lookup_name fails :-(", params->rank);
190 return -1;
191 }
192 }
193
194 // Connect...
195 int connect_retries = 0;
196 do
197 {
198 ret = MPI_Comm_connect(params->port_name, MPI_INFO_NULL, 0, MPI_COMM_WORLD, &(params->server));
199 if (MPI_SUCCESS != ret)
200 {
201 if (connect_retries == 0)
202 {
203 char cli_name[HOST_NAME_MAX];
204 gethostname(cli_name, HOST_NAME_MAX);
205 printf("----------------------------------------------------------------\n");
206 printf("XPN Client %s : Waiting for servers being up and runing...\n", cli_name);
207 printf("----------------------------------------------------------------\n\n");
208 }
209 connect_retries++;
210 sleep(2);
211 }
212 } while (MPI_SUCCESS != ret && connect_retries < 150);
213
214 if (MPI_SUCCESS != ret)
215 {
216 debug_error("Client[%d]: MPI_Comm_connect fails :-(", params->rank);
217 return -1;
218 }
219
220 debug_info("[CLI-COMM] end mpiClient_comm_connect(...)\n");
221
222 // Return OK
223 return 1;
224}

228 int ret;
229 int data;
230
231 debug_info("[CLI-COMM] begin mpiClient_comm_disconnect nservers\n");
232
233 data = MPI_SERVER_DISCONNECT;
234 int rank;
235 MPI_Comm_rank(MPI_COMM_WORLD, &(rank));
236 if (rank == 0)
237 {
238 MPI_Send(&data, 1, MPI_INT, 0, 0, params->server);
239 }
240
241 MPI_Barrier(MPI_COMM_WORLD);
242
243 // Disconnect
244 ret = MPI_Comm_disconnect(&(params->server));
245 if (MPI_SUCCESS != ret)
246 {
247 debug_error("Server[%d]: MPI_Comm_disconnect fails :-(", params->rank);
248 return -1;
249 }

251 // Return OK
