96/**
97 * @brief 'TODO'.
98 *
99 * 'TODO'.
100 *
101 * @par Parameters
102 * None.
103 * @par Returns
104 * Nothing.
105 */
106void fdstable_realloc(void)
107{
108 long old_size = fdstable_size;
109 struct generic_fd *fdstable_aux = fdstable;
110
111 if (NULL == fdstable)
112 {
113 fdstable_size = (long)MAX_FDS;
114 fdstable = (struct generic_fd *)malloc(fdstable_size * sizeof(struct generic_fd));
115 }
116 else
117 {
118 fdstable_size = fdstable_size * 2;
119 fdstable = (struct generic_fd *)realloc((struct generic_fd *)fdstable, fdstable_size * sizeof(struct generic_fd));
120 }
121
122 if (NULL == fdstable)
123 {
124 fprintf(stderr, "[bypass:%s:%d] Error: out of memory\n", __FILE__, __LINE__);
125 if (fdstable_aux != NULL)
126 {
127 free(fdstable_aux);
128 }
129 exit(-1);
130 }
131
132 for (int i = old_size; i < fdstable_size; ++i)
133 {
134 fdstable[i].type = FD_FREE;
135 fdstable[i].real_fd = -1;
136 fdstable[i].is_file = -1;
137 }
138}

163struct generic_fd fdstable_get(int fd)

166
167 struct generic_fd ret;
168
169 if (fd >= PLUSXPN)

172 ret = fdstable[fd];
173 }
174 else
175 {
176 ret.type = FD_SYS;
177 ret.real_fd = fd;
178 }
179
180 return ret;

183/**
184 * @brief 'TODO'.
185 *
186 * 'TODO'.
187 *
188 * @param fd 'TODO'.
189 * @return 'TODO'.
190 */
191int fdstable_put(struct generic_fd fd)
192{
193 for (int i = fdstable_first_free; i < fdstable_size; ++i)
194 {
195 if (fdstable[i].type == FD_FREE)
196 {
197 fdstable[i] = fd;
198 fdstable_first_free = (long)(i + 1);
199 // debug_info("[bypass] PUT FSTABLE %d %d %d\n", i, fdstable[i].type, fdstable[i].real_fd);
200 return i + PLUSXPN;
201 }
202 }
203
204 long old_size = fdstable_size;
205
206 fdstable_realloc();
207
208 if (fdstable[old_size].type == FD_FREE)
209 {
210 fdstable[old_size] = fd;
211 // debug_info("[bypass] PUT FSTABLE %d %d %d\n", i, fdstable[i].type, fdstable[i].real_fd);
212 return old_size + PLUSXPN;
213 }
214
215 return -1;
216}

226int add_xpn_file_to_fdstable(int fd)

230

382/**
383 * @brief 'TODO'.
384 *
385 * 'TODO'.
386 *
387 * @param dir 'TODO'.
388 * @return 'TODO'.
389 */
390int fdsdirtable_put(DIR *dir)
391{
392 // preparing the "file side" of the directory
393 struct generic_fd virtual_fd;
394 int fd;
395 int vfd;
396
397 fd = dirfd(dir);
398
399 virtual_fd.type = FD_XPN;
400 virtual_fd.real_fd = fd;
401 virtual_fd.is_file = 0;
402
403 // insert into the dirtable (and fdstable)
404 for (int i = fdsdirtable_first_free; i < fdsdirtable_size; ++i)
405 {
406 if (fdsdirtable[i] == NULL)
407 {
408 fdsdirtable[i] = dir;
409 fdsdirtable_first_free = (long)(i + 1);
410
411 vfd = fdstable_put(virtual_fd);
412 dir->fd = vfd;
413
414 return 0;
415 }
416 }
417
418 long old_size = fdstable_size;
419 fdsdirtable_realloc();
420
421 if (fdsdirtable[old_size] == NULL)
422 {
423 fdsdirtable[old_size] = dir;
424 fdsdirtable_first_free = (long)(old_size + 1);
425
426 vfd = fdstable_put(virtual_fd);
427 dir->fd = vfd;
428
429 return 0;
430 }
431
432 return -1;
433}

469 * This function checks if expand has already been initialized.

471 *
472 * @par Parameters
473 * None.
474 * @return 'TODO'.

476int xpn_adaptor_keepInit(void)
477{
478 int ret;
479 char *xpn_adaptor_initCalled_env = NULL;
480
481 if (xpn_adaptor_initCalled_getenv == 0)
482 {
483 xpn_adaptor_initCalled_env = getenv("INITCALLED");
484 xpn_adaptor_initCalled = 0;
485 if (xpn_adaptor_initCalled_env != NULL)
486 {
487 xpn_adaptor_initCalled = atoi(xpn_adaptor_initCalled_env);
488 }
489
490 xpn_adaptor_initCalled_getenv = 1;
491 }
492
493 ret = 0;
494 if (0 == xpn_adaptor_initCalled)
495 {
496 // If expand has not been initialized, then initialize it.
497
498 xpn_adaptor_initCalled = 1; // TODO: Delete
499 setenv("INITCALLED", "1", 1);
500
501 debug_info("[bypass]\t Before xpn_init()\n");
502 fdstable_init();
503 fdsdirtable_init();
504 ret = xpn_init();
505 debug_info("[bypass]\t After xpn_init() -> %d\n", ret);
506
507 if (ret < 0)
508 {
509 fprintf(stderr, "ERROR: Expand xpn_init couldn't be initialized :-(\n");
510 xpn_adaptor_initCalled = 0;
511 setenv("INITCALLED", "0", 1);
512 }
513 else
514 {
515 xpn_adaptor_initCalled = 1;
516 setenv("INITCALLED", "1", 1);
517 }
518 }
519
520 return ret;
521}

527int open(const char *path, int flags, ...)
528{
529 int ret, fd;
530 va_list ap;
531 mode_t mode = 0;
532
533 va_start(ap, flags);
534
535 mode = va_arg(ap, mode_t);
536
537 debug_info("[bypass] >> Before open....\n");
538 debug_info("[bypass] 1) Path => %s\n", path);
539 debug_info("[bypass] 2) Flags => %d\n", flags);
540 debug_info("[bypass] 3) Mode => %d\n", mode);
541
542 // This if checks if variable path passed as argument starts with the expand prefix.
543 if (is_xpn_prefix(path))
544 {
545 // We must initialize expand if it has not been initialized yet.
546 xpn_adaptor_keepInit();
547
548 // It is an XPN partition, so we redirect the syscall to expand syscall
549 debug_info("[bypass]\t xpn_open (%s,%o)\n", path + strlen(xpn_adaptor_partition_prefix), flags);
550
551 if (mode != 0)
552 {
553 fd = xpn_open(skip_xpn_prefix(path), flags, mode);
554 }
555 else
556 {
557 fd = xpn_open(skip_xpn_prefix(path), flags);
558 }
559
560 debug_info("[bypass]\t xpn_open (%s,%o) -> %d\n", skip_xpn_prefix(path), flags, fd);
561
562 ret = add_xpn_file_to_fdstable(fd);
563 }
564 // Not an XPN partition. We must link with the standard library.
565 else
566 {
567 debug_info("[bypass]\t dlsym_open (%s,%o,%o)\n", path, flags, mode);
568 ret = dlsym_open2((char *)path, flags, mode);
569 debug_info("[bypass]\t dlsym_open (%s,%o,%o) -> %d\n", path, flags, mode, ret);
570 }
571 va_end(ap);
572
573 debug_info("[bypass] << After open.... %s\n", path);
574 return ret;
575}
