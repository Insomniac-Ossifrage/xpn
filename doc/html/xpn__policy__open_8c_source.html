46void XpnGetURLServer(struct nfi_server *serv, char *abs_path, char *url_serv)
47{
48 char dir[PATH_MAX];
49 char dir_aux[PATH_MAX];
50 int abs_path_len;
51 int put_slash;
52 int pos_abs_path;
53 int pos_dir;
54
55 memccpy(dir, abs_path, 0, PATH_MAX - 1);
56 abs_path_len = strlen(abs_path);
57 put_slash = 1;
58 pos_abs_path = 0;
59 pos_dir = 0;
60
61 while (pos_abs_path < abs_path_len)
62 {
63 if (abs_path[pos_abs_path] == '/')
64 {
65 if (put_slash)
66 {
67 dir_aux[pos_dir] = abs_path[pos_abs_path];
68 pos_dir++;
69 put_slash = 0;
70 }
71 }
72 else
73 {
74 dir_aux[pos_dir] = abs_path[pos_abs_path];
75 pos_dir++;
76 put_slash = 1;
77 }
78 pos_abs_path++;
79 }
80 dir_aux[pos_dir] = '\0';
81 strcpy(dir, dir_aux);
82 sprintf(url_serv, "%s%s", serv->url, dir);
83}

85/**
86 * TODO:
87 * ** int op -> not used (!)
88 * ** XpnGetServers(.... int type==XPN_META_SERVER) -> XpnGetServers_mdata(....) ;
89 * ** fd < 0 => XpnGetServers_data_by_path(....) ;
90 * ** fd > 0 => XpnGetServers_data_by_fd(....) ;
91 *
92 * Returns the data or metadata servers.
93 *
94 * @param op Flag operation.
95 * @param pd Partition id.
96 * @param abs_path Absolute path.
97 * @param fd File descriptor.
98 * @param servers [out] The data or metadata servers to be obtained.
99 * @param type Metadata or data server.
100 *
101 * @return The number of data or metadata servers on success or -1 on error,
102 */
103int XpnGetServers(int op, int pd, __attribute__((__unused__)) char *abs_path, int fd, struct nfi_server ***servers, int type)
104{
105 struct nfi_server **serv;
106 int i, j, n;
107
108 n = 1; // TODO: check default value of 1
109
110 /* params op and abs_path are not used in this version */
111 switch (op)
112 {
113 default:
114 if (pd >= 0)
115 {
116 i = 0;
117 while ((i < XPN_MAX_PART) && (xpn_parttable[i].id != pd))
118 {
119 i++;
120 }
121
122 if (i == XPN_MAX_PART)
123 {
124 return -1;
125 }
126
127 switch (type)
128 {
129 case XPN_DATA_SERVER:
130 n = xpn_parttable[i].data_nserv;
131 break;
132 case XPN_META_SERVER:
133 n = xpn_parttable[i].meta_nserv;
134 break;
135 }
136
137 serv = (struct nfi_server **)malloc(sizeof(struct nfi_server *) * n);
138 if (serv == NULL)
139 {
140 /* xpn_err()*/
141 return -1;
142 }
143
144 for (j = 0; j < n; j++)
145 {
146 switch (type)
147 {
148 case XPN_DATA_SERVER:
149 serv[j] = &(xpn_parttable[i].data_serv[j]);
150 break;
151 case XPN_META_SERVER:
152 serv[j] = &(xpn_parttable[i].meta_serv[j]);
153 break;
154 }
155 }
156 (*servers) = serv;
157 }
158 else if (fd >= 0)
159 {
160 if ((fd > XPN_MAX_FILE) || (xpn_file_table[fd] == NULL))
161 {
162 return -1;
163 }
164
165 switch (type)
166 {
167 case XPN_DATA_SERVER:
168 n = xpn_file_table[fd]->data_vfh->n_nfih;
169 break;
170 case XPN_META_SERVER:
171 n = xpn_file_table[fd]->meta_vfh->n_nfih;
172 break;
173 }
174
175 serv = (struct nfi_server **)malloc(sizeof(struct nfi_server *) * n);
176 if (serv == NULL)
177 {
178 /* xpn_err()*/
179 return -1;
180 }
181
182 for (j = 0; j < n; j++)
183 {
184 switch (type)
185 {
186 case XPN_DATA_SERVER:
187 serv[j] = &(xpn_file_table[fd]->part->data_serv[j]);
188 break;
189 case XPN_META_SERVER:
190 serv[j] = &(xpn_file_table[fd]->part->meta_serv[j]);
191 break;
192 }
193 }
194
195 (*servers) = serv;
196 }
197 else
198 {
199 return -1;
200 }
201 break;
202 }
203
204 return n;
205}

209int XpnCreateMetadata(struct xpn_metadata *mdata, int pd, char *path)
210{
211 int part_id = 0;
212
213 if (mdata == NULL)
214 {
215 return -1;
216 }
217
218 // TODO pd == xpn_parttable[i].id
219 while ((part_id < XPN_MAX_PART) && (xpn_parttable[part_id].id != pd))
220 {
221 part_id++;
222 }
223
224 if (part_id == XPN_MAX_PART)
225 {
226 return -1;
227 }
228
229 /* initial values */
230 bzero(mdata, sizeof(struct xpn_metadata));
231 mdata->meta_nserv = xpn_parttable[part_id].meta_nserv;
232 mdata->data_nserv = xpn_parttable[part_id].data_nserv;
233 mdata->id = 0;
234 mdata->version = 1;
235 mdata->type = 0;
236 mdata->block_size = xpn_parttable[part_id].block_size;
237 mdata->type_policy = xpn_parttable[part_id].type;
238
239 if (mdata->type_policy < 0)
240 {
241 return -1;
242 }
243
244 switch (mdata->type_policy)
245 {
246 default:
247 mdata->policy = (struct policy *)malloc(sizeof(struct policy));
248 if (NULL == mdata->policy)
249 {
250 return -1;
251 }
252
253 ((struct policy *)mdata->policy)->first_node = hash(path, xpn_parttable[part_id].data_nserv);
254
255 mdata->policy_size = sizeof(struct policy);
256 break;
257 }
258 return 0;
259}

439int XpnGetAtribFd(int fd, struct stat *st)
440{
441 int ret, res, i, n, err;
442 struct nfi_server **servers;
443 struct nfi_attr *attr;
444
445 XPN_DEBUG_BEGIN_CUSTOM("%d", fd)
446
447 servers = NULL;
448 n = XpnGetServers(op_xpn_getattr, xpn_file_table[fd]->part->id, NULL, fd, &servers, XPN_DATA_SERVER);
449 if (n <= 0)
450 {
451 XPN_DEBUG_END_CUSTOM("%d", fd)
452 return -1;
453 }
454
455 attr = (struct nfi_attr *)malloc(n * sizeof(struct nfi_attr));
456 if (NULL == attr)
457 {
458 free(servers);
459 perror("malloc: ");
460 return -1;
461 }
462
463 memset(attr, 0, n * sizeof(struct nfi_attr));
464 memset(st, 0, sizeof(struct stat));
465
466 for (i = 0; i < n; i++)
467 {
468 res = XpnGetFh(xpn_file_table[fd]->mdata, &(xpn_file_table[fd]->data_vfh->nfih[i]), servers[i], xpn_file_table[fd]->path);
469 if (res < 0)
470 {
471 XPN_DEBUG_END_CUSTOM("%d", fd)
472 return res;
473 }
474
475 // Worker
476 nfi_worker_do_getattr(servers[i]->wrk, xpn_file_table[fd]->data_vfh->nfih[i], &(attr[i]));
477 }
478
479 // Wait
480 err = 0;
481 for (i = 0; i < n; i++)
482 {
483 ret = nfiworker_wait(servers[i]->wrk);
484
485 // Control error
486 if ((ret < 0) && (!err))
487 {
488 // erase the file create before de server number i
489 err = 1;
490 }
491 }
492
493 // Error checking
494 if (err)
495 {
496 xpn_err(XPNERR_REMOVE);
497 free(servers);
498 return -1;
499 }
500
501 st->st_size = 0;
502 for (i = 0; i < n; i++)
503 {
504 if (attr[i].at_size > 0)
505 {
506 st->st_size += (attr[i].at_size - XPN_HEADER_SIZE); // total size, in bytes
507 }
508
509 if (1 == attr[i].at_type) // It is a directory
510 {
511 break;
512 }
513 }
514
515 int master_node = ((struct policy *)xpn_file_table[fd]->mdata->policy)->first_node;
516 // int master_node = ((struct policy *)mdata->policy)->first_node;
517
518 st->st_dev = attr[master_node].st_dev; // device
519 st->st_ino = attr[master_node].st_ino; // inode
520
521 st->st_mode = attr[master_node].at_mode; // protection
522
523 if (0 == attr[master_node].at_type)
524 { // It is a file
525 st->st_mode = S_IFREG | st->st_mode;
526 }
527 if (1 == attr[master_node].at_type)
528 { // It is a directory
529 st->st_mode = S_IFDIR | st->st_mode;
530 }
531
532 st->st_nlink = attr[master_node].at_nlink; // number of hard links
533 st->st_uid = getuid(); // user ID of owner
534 st->st_gid = getgid(); // group ID of owner
535 // st->st_blksize = xpn_file_table[pd]->block_size ; /* blocksize for filesystem I/O // TODO
536 st->st_blocks = attr[master_node].at_blocks; // number of blocks allocated
537 st->st_atime = attr[master_node].at_atime; // time of last access
538 st->st_mtime = attr[master_node].at_mtime; // time of last modification
539 st->st_ctime = attr[master_node].at_ctime; // time of last change
540
541 free(servers);
542 free(attr);
543
544 return ret;
545}
