37void get_perm(int attr, char *attrstr)
38{
39 attrstr[0] = S_ISDIR(attr) ? 'd' : '-';
40 attrstr[1] = (attr & S_IRUSR) ? 'r' : '-';
41 attrstr[2] = (attr & S_IWUSR) ? 'w' : '-';
42 attrstr[3] = (attr & S_IXUSR) ? 'x' : '-';
43 attrstr[4] = (attr & S_IRGRP) ? 'r' : '-';
44 attrstr[5] = (attr & S_IWGRP) ? 'w' : '-';
45 attrstr[6] = (attr & S_IXGRP) ? 'x' : '-';
46 attrstr[7] = (attr & S_IROTH) ? 'r' : '-';
47 attrstr[8] = (attr & S_IWOTH) ? 'w' : '-';
48 attrstr[9] = (attr & S_IXOTH) ? 'x' : '-';
49 attrstr[10] = '\0';
50}

52void del_slash(char *path)
53{
54 int i, j, k;
55 char pathAux[NFSMAXPATHLEN];
56
57 i = 0;
58 j = 0;
59 k = 0;
60 while (path[i] != '\0')
61 {
62 if ((k == 0) && (path[i] == '/'))
63 {
64 pathAux[j] = path[i];
65 j++;
66 k = 1;
67 }
68 else if (path[i] != '/')
69 {
70 k = 0;
71 pathAux[j] = path[i];
72 j++;
73 }
74 i++;
75 }
76 pathAux[j] = '\0';
77 strcpy(path, pathAux);
78}
