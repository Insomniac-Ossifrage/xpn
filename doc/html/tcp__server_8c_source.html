83void tcp_server_dispatcher(struct st_th th)
84{
85 int ret;
86 int disconnect;
87 struct st_th th_arg;
88
89 // check params...
90 if (NULL == th.params)
91 {
92 printf("[WORKERS ID=%d] ERROR: NULL arguments", th.id);
93 return;
94 }
95
96 disconnect = 0;
97 while (!disconnect)
98 {
99 ret = tcp_server_comm_read_operation(th.params, (int)th.sd, (char *)&(th.type_op), 1, &(th.rank_client_id));
100
101 if (ret < 0)
102 {
103 debug_info("[TCP-SERVER] ERROR: tcp_server_comm_readdata fail\n");
104 return;
105 }
106
107 if (ret == 0)
108 {
109 debug_info("[TCP-SERVER] WARNING: tcp_server_comm_readdata broken pipe\n");
110 return;
111 }
112
113 if (th.type_op == TCP_SERVER_DISCONNECT)
114 {
115 debug_info("[TCP-SERVER] INFO: DISCONNECT received\n");
116 disconnect = 1;
117 continue;
118 }
119
120 if (th.type_op == TCP_SERVER_FINALIZE)
121 {
122 debug_info("[TCP-SERVER] INFO: FINALIZE received\n");
123 disconnect = 1;
124 the_end = 1;
125 continue;
126 }
127
128 // Launch worker per operation
129 th_arg.params = &params;
130 th_arg.sd = (int)th.sd;
131 th_arg.function = tcp_server_run;
132 th_arg.type_op = th.type_op;
133 th_arg.rank_client_id = th.rank_client_id;
134 th_arg.wait4me = FALSE;
135
136 tcp_server_run(th_arg);
137 }
138
139 debug_info("[TCP-SERVER] tcp_server_worker_run (ID=%d) close\n", th.rank_client_id);
140
141 tcp_server_comm_close((int)th.sd);
142}

249int tcp_server_down(void)
250{
251 int ret, sd, data;
252 char srv_name[1024];
253 char server_name[1024];
254 char port_number[1024];
255 FILE *file;
256
257 // Feedback
258 printf("\n");
259 printf(" -------------------\n");
260 printf(" Stopping servers... (%s)\n", serv_name);
261 printf(" -------------------\n");
262 printf("\n");
263
264 // Initialize
265 debug_msg_init();
266 ret = tcp_server_comm_init(&params);
267 if (ret < 0)
268 {
269 printf("[TCP-SERVER] ERROR: tcp_comm initialization fails\n");
270 return -1;
271 }
272 ret = workers_init(&worker, params.thread_mode);
273 if (ret < 0)
274 {
275 printf("[TCP-SERVER] ERROR: workers initialization fails\n");
276 return -1;
277 }
278
279 // Open host file
280 file = fopen(params.dns_file, "r");
281 if (file == NULL)
282 {
283 printf("[TCP-SERVER] ERROR: invalid file %s\n", params.shutdown_file);
284 return -1;
285 }
286
287 while (fscanf(file, "%s %s %s", srv_name, server_name, port_number) != EOF)
288 {
289 // Lookup port name
290 ret = ns_lookup("tcp_server", srv_name, server_name, port_number);
291 if (ret < 0)
292 {
293 printf("[TCP-SERVER] ERROR: server %s %s %s not found\n", srv_name, server_name, port_number);
294 continue;
295 }
296
297 // Connect with server
298 sd = tcp_server_comm_connect(&params, server_name, atoi(port_number));
299 if (sd < 0)
300 {
301 printf("[TCP-SERVER] ERROR: connect to %s failed\n", server_name);
302 continue;
303 }
304
305 // Send shutdown request
306 data = TCP_SERVER_FINALIZE;
307 ret = tcp_server_comm_write_data(&params, sd, (char *)&data, sizeof(int), 0); // 0: rank_client_id
308 if (ret < 0)
309 {
310 printf("[TCP-SERVER] ERROR: write SERVER_FINALIZE to %s failed\n", srv_name);
311 return -1;
312 }
313
314 // Close
315 tcp_server_comm_close(sd);
316 }
317
318 // Close host file
319 fclose(file);
320
321 // Wait and finalize for all current workers
322 debug_info("[TCP-SERVER] workers_destroy\n");
323 workers_destroy(&worker);
324 debug_info("[TCP-SERVER] tcp_server_comm_destroy\n");
325 tcp_server_comm_destroy(&params);
326
327 return 0;
328}
