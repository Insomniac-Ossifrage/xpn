146 break;
147
148 // File system

151 ret = wrk->server->ops->nfi_statfs(wrk->server, wrk->arg.inf);
152 break;
153 case op_preload:
154 debug_info("[%s] nfi_worker_run(%lu) -> nfi_preload\n", __FILE__, (unsigned long int)pthread_self());

160 break;
161 }
162
163 debug_info("[%s] nfi_worker_run(%lu) -> end op: %d\n", __FILE__, (unsigned long int)pthread_self(), wrk->arg.operation);
164
165 DEBUG_END();
166
167 wrk->arg.result = ret;
168}
