124int xpn_init_partition(__attribute__((__unused__)) char *partname)
125{
126 int res;
127 int i, j;
128 struct conf_connect_st *fd;
129 static int xpn_initialize = 0;
130 char *env_debug;
131
132 env_debug = getenv("XPN_DEBUG");
133 if ((env_debug != NULL) && (strlen(env_debug) > 0))
134 {
135 xpn_debug = 1;
136 }
137
138 XPN_DEBUG_BEGIN
139
140 setbuf(stdout, NULL);
141 setbuf(stderr, NULL);
142
143 pthread_mutex_lock(&xpn_init_mutex);
144
145 if (!xpn_initialize)
146 {
147 XPN_DEBUG("Initializing\n");
148 }
149 else
150 {
151 XPN_DEBUG("Already initialized\n");
152
153 pthread_mutex_unlock(&xpn_init_mutex);
154
155 res = 0;
156 XPN_DEBUG_END
157 return res;
158 }
159
160 fd = XpnPartitionOpen(); /* XpnPartitionOpen */
161 if (fd == NULL)
162 {
163 xpn_err(XPNERR_CONF);
164 fprintf(stderr, "xpn_init: Can't open partition data\n");
165
166 pthread_mutex_unlock(&xpn_init_mutex);
167
168 res = -1;
169 XPN_DEBUG_END
170 return res;
171 }
172
173 i = 0;
174 while ((res = XpnGetNextPartition(fd, xpn_parttable[i].name)) > 0)
175 {
176 xpn_parttable[i].initialized = 0;
177
178 xpn_parttable[i].id = XpnGetIdPartition(fd, xpn_parttable[i].name);
179 XPN_DEBUG("Partition %d: name=%s\n", xpn_parttable[i].id, xpn_parttable[i].name);
180
181 /* compruebo los errores???? */
182 if (XpnGetInfoPartition(fd, &(xpn_parttable[i])) == -1)
183 {
184 fprintf(stderr, "(4)xpn_init: %s info incomplete.\n", xpn_parttable[i].name);
185
186 pthread_mutex_unlock(&xpn_init_mutex);
187
188 res = -1;
189 XPN_DEBUG_END
190 return res;
191 }
192
193 xpn_parttable[i].data_nserv = XpnGetNumServersPartition(fd, &(xpn_parttable[i]), XPN_DATA_SERVER);
194 XPN_DEBUG("Partition %d: data_nserv=%d\n", xpn_parttable[i].id, xpn_parttable[i].data_nserv);
195
196 xpn_parttable[i].data_serv = (struct nfi_server *)malloc(xpn_parttable[i].data_nserv * sizeof(struct nfi_server));
197 if (xpn_parttable[i].data_serv == NULL)
198 {
199 fprintf(stderr, "xpn_init: Couldn't allocate memory\n");
200
201 pthread_mutex_unlock(&xpn_init_mutex);
202
203 res = -1;
204 XPN_DEBUG_END
205 return res;
206 }
207
208 memset(xpn_parttable[i].data_serv, 0, xpn_parttable[i].data_nserv * sizeof(struct nfi_server));
209
210 for (j = 0; j < xpn_parttable[i].data_nserv; j++)
211 {
212 // TODO: AQUI??
213
214 res = XpnGetServer(fd, &(xpn_parttable[i]), &(xpn_parttable[i].data_serv[j]), XPN_DATA_SERVER);
215 if (res < 0)
216 {
217
218 XpnPartitionClose(fd);
219 for (j = 0; j < i; j++)
220 {
221 xpn_destroy_servers(&(xpn_parttable[j]));
222 }
223 fprintf(stderr, "xpn_init: Data configuration incorrect\n");
224
225 pthread_mutex_unlock(&xpn_init_mutex);
226
227 res = -1;
228 XPN_DEBUG_END
229 return res;
230 }
231 }
232 XPN_DEBUG("Partition %d end", xpn_parttable[i].id);
233
234 i++;
235 if (i == XPN_MAX_PART)
236 {
237 XpnPartitionClose(fd);
238 for (j = 0; j < XPN_MAX_PART; j++)
239 {
240 xpn_destroy_servers(&(xpn_parttable[j]));
241 }
242 fprintf(stderr, "xpn_init: Data configuration incorrect. Too much partitions \n");
243
244 pthread_mutex_unlock(&xpn_init_mutex);
245
246 res = -1;
247 XPN_DEBUG_END
248 return res;
249 }
250 } // while
251
252 if (res < 0)
253 {
254 XpnPartitionClose(fd);
255 for (j = 0; j < i; j++)
256 {
257 xpn_destroy_servers(&(xpn_parttable[j]));
258 }
259
260 fprintf(stderr, "xpn_init: Data configuration incorrect.\n");
261
262 pthread_mutex_unlock(&xpn_init_mutex);
263
264 res = -1;
265 XPN_DEBUG_END
266 return res;
267 }
268
269 XpnPartitionClose(fd);
270
271 /* Init the file table */
272 res = xpn_init_file_table();
273 if (res < 0)
274 {
275 for (j = 0; j < i; j++)
276 {
277 xpn_destroy_servers(&(xpn_parttable[j]));
278 }
279
280 pthread_mutex_unlock(&xpn_init_mutex);
281
282 res = -1;
283 XPN_DEBUG_END
284 return res;
285 }
286
287 /* Init the rest of elements of the table */
288 for (j = 0; j < i; j++)
289 {
290 xpn_parttable[i].id = -1;
291 }
292 xpn_init_cwd();
293 xpn_initialize = 1;
294
295 pthread_mutex_unlock(&xpn_init_mutex);
296
297 res = 0;
298 XPN_DEBUG_END
299
300 return res;
301}
