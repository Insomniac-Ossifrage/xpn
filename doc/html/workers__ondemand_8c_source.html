49void *worker_run(void *arg)
50{
51 struct st_th th;
52
53 DEBUG_BEGIN();
54
55 // get initial pointers...
56 struct st_th *th_aux = (struct st_th *)arg;
57 if (NULL == th_aux)
58 {
59 debug_info("ERROR: worker_run receive arg == NULL\n");
60 return NULL;
61 }
62 worker_ondemand_t *w_aux = (worker_ondemand_t *)th_aux->w;
63 if (NULL == w_aux)
64 {
65 debug_info("ERROR: worker_run receive arg->w == NULL\n");
66 return NULL;
67 }
68 struct st_th *th_shadow = (struct st_th *)(th_aux->v);
69 if (NULL == th_shadow)
70 {
71 debug_info("ERROR: worker_run receive arg->v == NULL\n");
72 return NULL;
73 }
74
75 // prolog: copy arguments and update n_workers
76 pthread_mutex_lock(&(w_aux->m_worker));
77 memcpy(&th, arg, sizeof(struct st_th));
78 w_aux->busy_worker = FALSE;
79 w_aux->n_workers++;
80 pthread_cond_broadcast(&(w_aux->c_worker)); // pthread_cond_signal(&c_worker);
81 pthread_mutex_unlock(&(w_aux->m_worker));
82
83 // do function code...
84 th.function(th);
85
86 // do wakeup worker_ondemand_wait(...) if needed
87 if (TRUE == th.wait4me)
88 {
89 pthread_mutex_lock(&(th_shadow->m_wait));
90 th_shadow->r_wait = FALSE;
91 pthread_cond_broadcast(&(th_shadow->c_wait));
92 pthread_mutex_unlock(&(th_shadow->m_wait));
93 }
94
95 // epilog: update n_workers
96 pthread_mutex_lock(&(w_aux->m_worker));
97 w_aux->n_workers--;
98 pthread_cond_broadcast(&(w_aux->c_nworkers)); // pthread_cond_signal(&c_nworkers);
99 pthread_mutex_unlock(&(w_aux->m_worker));
100
101 DEBUG_END();
102
103 // end
104 pthread_exit(0);
105 return NULL;
106}

128int worker_ondemand_launch(worker_ondemand_t *w, struct st_th *th_arg, void (*worker_function)(struct st_th))
129{
130 int ret;
131 pthread_attr_t th_attr;
132 pthread_t th_worker;
133 static int th_cont = 0;
134
135 DEBUG_BEGIN();
136
137 pthread_attr_init(&th_attr);
138 pthread_attr_setdetachstate(&th_attr, PTHREAD_CREATE_DETACHED);
139 pthread_attr_setstacksize(&th_attr, STACK_SIZE);
140 w->busy_worker = TRUE;
141
142 // prepare arguments...
143 th_arg->id = th_cont++;
144 th_arg->function = worker_function;
145 th_arg->w = (void *)w;
146 th_arg->v = (void *)th_arg;
147
148 // create thread...
149 debug_info("[WORKERS] pthread_create: create_thread worker_run\n");
150 ret = pthread_create(&th_worker, &th_attr, (void *(*)(void *))(worker_run), (void *)th_arg);
151 if (ret != 0)
152 {
153 debug_error("[WORKERS] pthread_create %d\n", ret);
154 perror("pthread_create: Error en create_thread: ");
155 }
156
157 // wait to copy args...
158 debug_info("[WORKERS] pthread_create: lock worker_run\n");
159 pthread_mutex_lock(&(w->m_worker));
160 while (w->busy_worker == TRUE)
161 {
162 debug_info("[WORKERS] pthread_create: wait worker_run\n");
163 pthread_cond_wait(&(w->c_worker), &(w->m_worker));
164 }
165
166 debug_info("[WORKERS] pthread_create: busy_worker= TRUE worker_run\n");
167 w->busy_worker = TRUE;
168 debug_info("[WORKERS] pthread_create: unlock worker_run\n");
169 pthread_mutex_unlock(&(w->m_worker));
170
171 DEBUG_END();
172 return 0;
173}

194{
195 DEBUG_BEGIN();
196
197 // wait to n_workers be zero...
198 debug_info("[WORKERS] pthread_create: lock workers_ondemand_wait\n");
199 pthread_mutex_lock(&(w->m_worker));

202 debug_info("[WORKERS] pthread_create: wait workers_ondemand_wait\n");
203 pthread_cond_wait(&(w->c_nworkers), &(w->m_worker));

205 debug_info("[WORKERS] pthread_create: unlock workers_ondemand_wait\n");
206 pthread_mutex_unlock(&(w->m_worker));
207
208 // destroy resources
209 pthread_cond_destroy(&(w->c_worker));
210 pthread_cond_destroy(&(w->c_nworkers));
211 pthread_mutex_destroy(&(w->m_worker));

213 DEBUG_END();
214}
