94void NFItoLOCALattr(struct stat *att, struct nfi_attr *nfi_att)
95{
96 att->st_dev = nfi_att->st_dev;
97 att->st_ino = nfi_att->st_ino;
98
99 if (nfi_att->at_type == NFIFILE)
100 {
101 att->st_mode = nfi_att->at_mode | S_IFREG; // protection
102 }
103
104 if (nfi_att->at_type == NFIDIR)
105 {
106 att->st_mode = nfi_att->at_mode | S_IFDIR; // protection
107 }
108
109 att->st_nlink = nfi_att->at_nlink; // number of hard links
110 att->st_uid = nfi_att->at_uid; // user ID of owner
111 att->st_gid = nfi_att->at_gid; // group ID of owner
112 att->st_size = nfi_att->at_size; // size
113 att->st_blksize = nfi_att->at_blksize; // blocksize for filesystem I/O
114 att->st_blocks = nfi_att->at_blocks; // number of blocks allocated
115 att->st_atime = nfi_att->at_atime; // time of last access
116 att->st_mtime = nfi_att->at_mtime; // time of last modification
117 att->st_ctime = nfi_att->at_ctime; // time of last change
118}

178int nfi_local_init(char *url, struct nfi_server *serv, __attribute__((__unused__)) struct nfi_attr_server *attr)
179{
180 int ret;
181 char server[PATH_MAX], dir[PATH_MAX], prt[PATH_MAX];
182 struct nfi_local_server *server_aux;
183
184 DEBUG_BEGIN();
185
186 // check params...
187 if (serv == NULL)
188 {
189 debug_error("ERROR: serv argument is NULL.\n");
190 return -1;
191 }
192
193 // Alloc for serv->ops...
194 serv->ops = (struct nfi_ops *)malloc(sizeof(struct nfi_ops));
195 NULL_RET_ERR(serv->ops, LOCALERR_MEMORY);
196
197 // Fill serv->ops...
198 bzero(serv->ops, sizeof(struct nfi_ops));
199 serv->ops->nfi_reconnect = nfi_local_reconnect;
200 serv->ops->nfi_disconnect = nfi_local_disconnect;
201
202 serv->ops->nfi_open = nfi_local_open;
203 serv->ops->nfi_create = nfi_local_create;
204 serv->ops->nfi_read = nfi_local_read;
205 serv->ops->nfi_write = nfi_local_write;
206 serv->ops->nfi_close = nfi_local_close;
207 serv->ops->nfi_remove = nfi_local_remove;
208 serv->ops->nfi_rename = nfi_local_rename;
209 serv->ops->nfi_getattr = nfi_local_getattr;
210 serv->ops->nfi_setattr = nfi_local_setattr;
211
212 serv->ops->nfi_opendir = nfi_local_opendir;
213 serv->ops->nfi_mkdir = nfi_local_mkdir;
214 serv->ops->nfi_readdir = nfi_local_readdir;
215 serv->ops->nfi_closedir = nfi_local_closedir;
216 serv->ops->nfi_rmdir = nfi_local_rmdir;
217
218 // serv->ops->nfi_preload = nfi_local_preload;
219 // serv->ops->nfi_flush = nfi_local_flush;
220
221 serv->ops->nfi_statfs = nfi_local_statfs;
222
223 // ParseURL...
224 ret = ParseURL(url, prt, NULL, NULL, server, NULL, dir);
225 if (ret < 0)
226 {
227 // local_err(LOCALERR_URL) ;
228 debug_error("ERROR: nfi_local_init found incorrect URL '%s'.\n", url);
229 FREE_AND_NULL(serv->ops);
230 return -1;
231 }
232
233 // Alloc for serv_aux...
234 server_aux = (struct nfi_local_server *)malloc(sizeof(struct nfi_local_server));
235 if (server_aux == NULL)
236 {
237 debug_error("ERROR: out of memory\n");
238 FREE_AND_NULL(serv->ops);
239 return -1;
240 }
241
242 // Initialize serv_aux...
243 memset(server_aux, 0, sizeof(struct nfi_local_server));
244
245 serv->xpn_thread = TH_NOT;
246 char *env_thread = getenv("XPN_THREAD");
247 if (env_thread != NULL)
248 {
249 serv->xpn_thread = atoi(env_thread);
250 }
251
252 // copy 'url' string...
253 // serv->protocol = LOCAL; //TODO: delete
254 strcpy(server_aux->path, dir);
255 serv->private_info = (void *)server_aux;
256
257 serv->url = STRING_MISC_StrDup(url); // full URL
258 if (serv->url == NULL)
259 {
260 debug_error("ERROR: out of memory\n");
261 return -1;
262 }
263
264 serv->server = STRING_MISC_StrDup(server); // URL.server
265 if (serv->server == NULL)
266 {
267 debug_error("ERROR: out of memory\n");
268 return -1;
269 }
270
271 // nfi_worker...
272 serv->wrk = (struct nfi_worker *)malloc(sizeof(struct nfi_worker));
273 memset(serv->wrk, 0, sizeof(struct nfi_worker));
274 serv->wrk->server = serv;
275
276 debug_info("[NFI-LOCAL] nfiworker_init()\n");
277 ret = nfiworker_init(serv);
278
279 ret = nfi_local_connect(serv, url, prt, server, dir);
280 if (ret < 0)
281 {
282 FREE_AND_NULL(serv->ops);
283 FREE_AND_NULL(server_aux);
284 return -1;
285 }
286
287 DEBUG_END();
288
289 // Return OK
290 return 0;
291}

444int nfi_local_open(struct nfi_server *serv, char *url, struct nfi_fhandle *fho)
445{
446 int ret;
447 char dir[PATH_MAX];
448 struct nfi_local_fhandle *fh_aux;
449
450 DEBUG_BEGIN();
451
452 // Check arguments...
453 NULL_RET_ERR(serv, LOCALERR_PARAM);
454 NULL_RET_ERR(fho, LOCALERR_PARAM);
455 nfi_local_keepConnected(serv);
456 NULL_RET_ERR(serv->private_info, LOCALERR_PARAM);
457
458 // from url -> server + dir
459 ret = ParseURL(url, NULL, NULL, NULL, NULL, NULL, dir);
460 if (ret < 0)
461 {
462 fprintf(stderr, "nfi_local_open: url %s incorrect.\n", url);
463 local_err(LOCALERR_URL);
464 return -1;
465 }
466
467 fho->url = strdup(url);
468 NULL_RET_ERR(fho->url, LOCALERR_MEMORY);
469
470 fh_aux = (struct nfi_local_fhandle *)malloc(sizeof(struct nfi_local_fhandle));
471 if (fh_aux == NULL)
472 {
473 local_err(LOCALERR_MEMORY);
474 FREE_AND_NULL(fho->url);
475 return -1;
476 }
477
478 // Do open
479 fh_aux->fd = real_posix_open2(dir, O_RDWR, S_IRWXU);
480 if (fh_aux->fd < 0)
481 {
482 debug_error("real_posix_open fails to open '%s' in server %s.\n", dir, serv->server);
483 FREE_AND_NULL(fh_aux);
484 FREE_AND_NULL(fho->url);
485 return -1;
486 }
487
488 strcpy(fh_aux->path, dir);
489
490 fho->type = NFIFILE;
491 fho->priv_fh = NULL;
492 fho->server = serv;
493 fho->priv_fh = (void *)fh_aux;
494
495 DEBUG_END();
496
497 return 0;
498}

500int nfi_local_create(struct nfi_server *serv, char *url, struct nfi_attr *attr, struct nfi_fhandle *fh)
501{
502 int ret;
503 char dir[PATH_MAX];
504 struct nfi_local_fhandle *fh_aux;
505 struct stat st;
506
507 DEBUG_BEGIN();
508
509 // Check arguments...
510 NULL_RET_ERR(serv, LOCALERR_PARAM);
511 NULL_RET_ERR(attr, LOCALERR_PARAM);
512 nfi_local_keepConnected(serv);
513 NULL_RET_ERR(serv->private_info, LOCALERR_PARAM);
514
515 // url -> server + dir
516 ret = ParseURL(url, NULL, NULL, NULL, NULL, NULL, dir);
517 if (ret < 0)
518 {
519 fprintf(stderr, "ERROR: nfi_local_create: url %s incorrect.\n", url);
520 local_err(LOCALERR_URL);
521 return -1;
522 }
523
524 // private_info file handle
525 fh_aux = (struct nfi_local_fhandle *)malloc(sizeof(struct nfi_local_fhandle));
526 NULL_RET_ERR(fh_aux, LOCALERR_MEMORY);
527 bzero(fh_aux, sizeof(struct nfi_local_fhandle));
528
529 // Do creat
530 fh_aux->fd = real_posix_open2(dir, O_CREAT | O_RDWR | O_TRUNC, attr->at_mode);
531 if (fh_aux->fd < 0)
532 {
533 debug_error("files_posix_open fails to creat '%s' in server '%s'.\n", dir, serv->server);
534 FREE_AND_NULL(fh_aux);
535 return -1;
536 }
537
538 // Get stat of the file
539 memset(&st, 0, sizeof(struct stat));
540 ret = real_posix_stat(dir, &st);
541 if (ret < 0)
542 {
543 debug_error("real_posix_stat fails to stat '%s' in server '%s'.\n", fh_aux->path, serv->server);
544 FREE_AND_NULL(fh->url);
545 FREE_AND_NULL(fh_aux);
546 return -1;
547 }
548
549 fh->type = NFIFILE;
550 fh->server = serv;
551 fh->priv_fh = (void *)fh_aux;
552
553 fh->url = strdup(url);
554 if (fh->url == NULL)
555 {
556 local_err(LOCALERR_MEMORY);
557 FREE_AND_NULL(fh_aux);
558 return -1;
559 }
560
561 LOCALtoNFIattr(attr, &st);
562
563 DEBUG_END();
564
565 return 0;
566}
