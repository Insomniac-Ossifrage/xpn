40int do_cp(char *src_path, char *dest_path)
41{
42 int ret, fd_src, fd_dest;
43 int cont = 0;
44 int cont2 = 0;
45 int buf_len = BUFFER_SIZE;
46
47 fd_src = open(src_path, O_RDONLY);
48 if (fd_src < 0)
49 {
50 perror("open source: ");
51 return -1;
52 }
53
54 // TODO: mkdir any directory from dest_path before open it, just in case parent directories doesn't exists...
55
56 fd_dest = open(dest_path, O_WRONLY | O_CREAT | O_TRUNC, 0755);
57 if (fd_dest < 0)
58 {
59 perror("open destination: ");
60 return -1;
61 }
62
63 do
64 {
65 cont = 0;
66 buf_len = BUFFER_SIZE;
67 do
68 {
69 ret = read(fd_src, buf + cont, buf_len);
70 if (ret == -1)
71 {
72 perror("read: ");
73 return -1;
74 }
75
76 cont = cont + ret;
77 buf_len = buf_len - ret;
78 } while ((cont < buf_len) && (ret > 0));
79
80 cont2 = 0;
81 buf_len = cont;
82
83 do
84 {
85 ret = write(fd_dest, buf + cont2, buf_len);
86 if (ret == -1)
87 {
88 perror("write: ");
89 return -1;
90 }
91
92 cont2 = cont2 + ret;
93 buf_len = buf_len - ret;
94 } while ((cont2 < cont) && (ret > 0));
95
96 } while (cont > 0);
97
98 close(fd_src);
99 close(fd_dest);
100
101 return 0;
102}

104int do_cp_recursive(char *src_path, char *dest_path)
105{
106 struct stat stat_buf;
107 int ret = 0;
108
109 ret = stat(src_path, &stat_buf);
110 if (ret < 0)
111 {
112 perror("stat source: ");
113 printf("ERROR at '%s': stat failed\n", src_path);
114 return -1;
115 }
116
117 if (S_ISREG(stat_buf.st_mode))
118 {
119 printf("=");
120 // printf("cp '%s' '%s' ...\n", src_path, dest_path);
121 return do_cp(src_path, dest_path);
122 }
123 else if (S_ISDIR(stat_buf.st_mode))
124 {
125 DIR *dir = NULL;
126 struct dirent *entry = NULL;
127 char path_src[PATH_MAX];
128 char path_dst[PATH_MAX];
129
130 // mkdir in destination
131 // printf("mkdir '%s' ...\n", dest_path);
132 ret = mkdir(dest_path, stat_buf.st_mode);
133 if ((ret == -1) && (errno != EEXIST))
134 {
135 perror("mkdir:");
136 return -1;
137 }
138
139 // traverse a src tree...
140 dir = opendir(src_path);
141 if (NULL == dir)
142 {
143 perror("opendir:");
144 return -1;
145 }
146
147 entry = readdir(dir);
148 while (entry != NULL)
149 {
150 if (!strcmp(entry->d_name, "."))
151 {
152 entry = readdir(dir);
153 continue;
154 }
155 if (!strcmp(entry->d_name, ".."))
156 {
157 entry = readdir(dir);
158 continue;
159 }
160
161 sprintf(path_src, "%s/%s", src_path, entry->d_name);
162 sprintf(path_dst, "%s/%s", dest_path, entry->d_name);
163 do_cp_recursive(path_src, path_dst);
164
165 entry = readdir(dir);
166 }
167
168 closedir(dir);
169 }
170 else
171 {
172 printf("ERROR at '%s': not a directory or file\n", src_path);
173 }
174
175 return ret;
176}
